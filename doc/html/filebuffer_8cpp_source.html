68 void FileBuffer::loadFromFile(QFile &file)
69 {
70  _pos = 0;
71  _data.clear();
72  if (!file.open(QIODevice::ReadOnly))
73  throw FileReadError("unable to open file for reading");
74  _data.reserve(file.size());
75  QDataStream in(&file);
76  quint8 byte;
77  while (!in.atEnd()) {
78  in >> byte;
79  _data.push_back(byte);
80  }
81  file.close();
82  cout << "Read " << _data.size() << " bytes from '" << file.fileName().toStdString() << "'" << endl;
83 }
