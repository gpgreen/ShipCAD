142 static QVector3D CalculateFlowDirection2(QVector3D incoming, SubdivisionPoint* point)
143 {
144  incoming.normalize();
145  QVector3D normal = point->getNormal();
146  QVector3D p = point->getCoordinate();
147  Plane plane(p, normal);
148  QVector3D direction = normal + incoming;
149  direction.normalize();
150  incoming = p + direction;
151  QVector3D proj = plane.projectPointOnPlane(incoming);
152  direction = proj - p;
153  direction.normalize();

204 static size_t FindInitialTriangle(bool method_new,
205  vector<Triangle>& triangles,
206  QVector3D startpoint, QVector3D endpoint,
207  QVector3D& intersection, QVector3D& direction)
208 {
209  size_t result = triangles.size();
210  double distance = 1E8;
211  intersection = ZERO;
212  direction = ZERO;
213  for (size_t i=0; i<triangles.size(); ++i) {
214  Triangle& triangle = triangles[i];
215  float s1 = triangle.plane.distance(startpoint);
216  float s2 = triangle.plane.distance(endpoint);
217  if ((s1 < 0 && s2 > 0) || (s1 > 0 && s2 < 0)) {
218  // possible intersection
219  float t;
220  if (s1 == s2)
221  t = 0.5;
222  else
223  t = -s1 / (s2 - s1);
224  QVector3D p = startpoint + t * (endpoint - startpoint);
225  if (PointInTriangle(p, triangle.p1->second.coord,
226  triangle.p2->second.coord,
227  triangle.p3->second.coord)) {
228  t = startpoint.distanceToPoint(p);
229  if (t < distance) {
230  distance = t;
231  result = i;
232  intersection = p;
233  }
234  }
235  }
236  }
237  if (result != triangles.size()) {
238  // Calculate baycentric coordinates to interpolate between the three flowdirections
239  // http://softsurfer.com/Archive/algorithm_0104/algorithm_0104.htm
240  Triangle& triangle = triangles[result];
241  QVector3D p0 = triangle.p1->second.coord;
242  QVector3D p1 = triangle.p2->second.coord;
243  QVector3D p2 = triangle.p3->second.coord;
244  QVector3D u = p1 - p0;
245  QVector3D v = p2 - p0;
246  QVector3D w = intersection - p0;
247  double udotu = QVector3D::dotProduct(u, u);
248  double udotv = QVector3D::dotProduct(u, v);
249  double vdotv = QVector3D::dotProduct(v, v);
250  double wdotu = QVector3D::dotProduct(w, u);
251  double wdotv = QVector3D::dotProduct(w, v);
252  double s = (udotv*wdotv-vdotv*wdotu)/(udotv*udotv-udotu*vdotv);
253  double t = (udotv*wdotu-udotu*wdotv)/(udotv*udotv-udotu*vdotv);
254  double b0 = 1 - s - t;
255  double b1 = s;
256  double b2 = t;
257  // check
258  t = b0 + b1 + b2;
259  if (t == 1) {
260  p0 = triangle.p1->second.flowdir;
261  p1 = triangle.p2->second.flowdir;
262  p2 = triangle.p3->second.flowdir;
263  direction = b0 * p0 + b1 * p1 + b2 * p2;
264 
265  if (method_new) {
266  direction = QVector3D(-1, 0.1, -0.1);
267  direction.normalize();
268  }
269  }
270  }
271  return result;
272 }

293 // used in rebuild
294 // returns true if triangle is valid
295 // nextindex will be set to the next triangle to process, or to maxindex if no more to process
296 // skipind1 and skipind2 are set to the points to ignore for intersections, will be updated
297 // in this method with the new points to ignore if valid
298 static bool ProcessTriangle(bool method_new, Triangle& triangle,
299  map<SubdivisionPoint*, PointData>::iterator& skipind1,
300  map<SubdivisionPoint*, PointData>::iterator& skipind2,
301  QVector3D& intersection,
302  QVector3D& direction,
303  size_t& nextindex, size_t maxindex)
304 {
305  bool result = false;
306  nextindex = triangle.index;
307  triangle.processed = true;
308  QVector3D p1 = triangle.plane.projectPointOnPlane(intersection);
309  p1 = p1 + 0.0005 * direction;
310  if (!PointInTriangle(p1, triangle.p1->second.coord, triangle.p2->second.coord,
311  triangle.p3->second.coord)) {
312  p1 = triangle.plane.projectPointOnPlane(intersection);
313  }
314 
315  p1 = triangle.plane.projectPointOnPlane(p1);
316  double distance = 50;
317  map<SubdivisionPoint*, PointData>::iterator ind1;
318  map<SubdivisionPoint*, PointData>::iterator ind2;
319  QVector3D p2 = p1 + distance * direction;
320  // test all three linesegments for intersection
321  for (size_t i=0; i<3; ++i) {
322  if (i == 0) {
323  ind1 = triangle.p1;
324  ind2 = triangle.p2;
325  } else if (i == 1) {
326  ind1 = triangle.p2;
327  ind2 = triangle.p3;
328  } else {
329  ind1 = triangle.p3;
330  ind2 = triangle.p1;
331  }
332  QVector3D int1;
333  double param;
334  if ((ind1 == skipind1 && ind2 == skipind2) || (ind1 == skipind2 && ind2 == skipind1)) {
335  // does nothing
336  } else if (Lines3DIntersect(p1, p2, ind1->second.coord, ind2->second.coord, param, int1)) {
337  distance = triangle.plane.distance(int1);
338  if (distance < 1E-1) {
339  intersection = int1;
340  QVector3D dir1;
341  QVector3D dir2;
342  // calculate direction
343  if (method_new) {
344  dir1 = CalculateFlowDirection2(direction, ind1->first);
345  dir2 = CalculateFlowDirection2(direction, ind2->first);
346  } else {
347  dir1 = ind1->second.flowdir;
348  dir2 = ind2->second.flowdir;
349  }
350  skipind1 = ind1;
351  skipind2 = ind2;
352  direction = dir1 + param * (dir2 - dir1);
353  nextindex = NextTriangleIndex(ind1, ind2, triangle.index, maxindex);
354  result = true;
355  break;
356  }
357  }
358  }
359 
360  return result;
361 }

364 void Flowline::rebuild()
365 {
366  // clear spline data
367  Spline::clear();
368 
369  // assemble all faces that are (partially) submerged and extract points
370  float wlheight = _owner->findLowestHydrostaticsPoint() + _owner->getProjectSettings().getDraft();
371 
372  SubdivisionSurface* surf = _owner->getSurface();
373  if (surf->numberOfPoints() == 0)
374  return;
375  vector<SubdivisionFace*> faces;
376  for (size_t i=0; i<surf->numberOfLayers(); ++i) {
377  if (!surf->getLayer(i)->useInHydrostatics())
378  continue;
379  SubdivisionLayer* layer = surf->getLayer(i);
380  for (size_t j=0; j<layer->numberOfFaces(); ++j) {
381  if (layer->getFace(j)->getMin().z() > wlheight)
382  continue;
383  SubdivisionControlFace* face = layer->getFace(j);
384  for (size_t k=0; k<face->numberOfChildren(); ++k) {
385  SubdivisionFace* child = face->getChild(k);
386  for (size_t l=0; l<child->numberOfPoints(); ++l) {
387  if (child->getPoint(l)->getCoordinate().z() <= wlheight) {
388  // face is (partially) submerged
389  faces.push_back(child);
390  break;
391  }
392  }
393  }
394  }
395  }
396  if (faces.size() == 0) {
397  setBuild(true);
398  return;
399  }
400  set<SubdivisionPoint*> points;
401  for (size_t i=0; i<faces.size(); ++i) {
402  SubdivisionFace* child = faces[i];
403  for (size_t j=0; j<child->numberOfPoints(); ++j) {
404  SubdivisionPoint* point = child->getPoint(j);
405  points.insert(point);
406  }
407  }
408  map<SubdivisionPoint*, PointData> pointdata;
409  for (set<SubdivisionPoint*>::iterator i=points.begin(); i!=points.end(); i++) {
410  PointData pd;
411  pd.coord = (*i)->getCoordinate();
412  pd.flowdir = CalculateFlowDirection(*i);
413  pointdata.insert(make_pair(*i, pd));
414  }
415  vector<Triangle> triangles;
416  for (size_t i=0; i<faces.size(); ++i) {
417  SubdivisionFace* child = faces[i];
418  for (size_t j=2; j<child->numberOfPoints(); ++j)
419  AddTriangle(triangles, pointdata, child->getPoint(0), child->getPoint(j-1), child->getPoint(j));
420  }
421  QVector3D startpoint;
422  QVector3D endpoint;
423  switch(_projection_vw) {
424  case fvProfile:
425  startpoint.setX(_projection_point.x());
426  startpoint.setY(_owner->getSurface()->getMax().y() + 10);
427  startpoint.setZ(_projection_point.y());
428  endpoint = QVector3D(startpoint.x(), 0, startpoint.z());
429  break;
430  case fvPlan:
431  startpoint.setX(_projection_point.x());
432  startpoint.setY(_projection_point.y());
433  startpoint.setZ(_owner->getSurface()->getMin().z() - 10);
434  endpoint = QVector3D(startpoint.x(), startpoint.z(),
435  _owner->getSurface()->getMax().z() + 100);
436  break;
437  case fvBodyplan:
438  if (_projection_point.x() < 0) {
439  startpoint.setX(_owner->getSurface()->getMin().x() - 10);
440  startpoint.setY(-_projection_point.x());
441  startpoint.setZ(_projection_point.y());
442  endpoint = QVector3D(_owner->getSurface()->getMax().x() + 10,
443  startpoint.y(), startpoint.z());
444  } else {
445  startpoint.setX(_owner->getSurface()->getMax().x() + 10);
446  startpoint.setY(_projection_point.x());
447  startpoint.setZ(_projection_point.y());
448  endpoint = QVector3D(_owner->getSurface()->getMin().x() - 10,
449  startpoint.y(), startpoint.z());
450  }
451  break;
452  default:
453  setBuild(true);
454  return;
455  };
456 
457  // find the initial triangle
458  QVector3D intersection;
459  QVector3D direction;
460  size_t index = FindInitialTriangle(_method_new, triangles,
461  startpoint, endpoint, intersection, direction);
462  map<SubdivisionPoint*, PointData>::iterator skipind1 = pointdata.end();
463  map<SubdivisionPoint*, PointData>::iterator skipind2 = pointdata.end();
464  if (index != triangles.size()) {
465  add(intersection);
466  // trace triangles from here
467  size_t iteration = 0;
468  bool valid;
469  do {
470  if (triangles[index].processed)
471  valid = false;
472  else
473  valid = ProcessTriangle(_method_new, triangles[index], skipind1, skipind2,
474  intersection, direction, index, triangles.size());
475  if (valid) {
476  add(intersection);
477  }
478 
479  else {
480  valid = ProcessTriangle(_method_new, triangles[index], skipind1, skipind2,
481  intersection, direction, index, triangles.size());
482  }
483  ++iteration;
484  } while (valid && index != triangles.size() && iteration < 5000);
485  while (numberOfPoints() > 1) {
486  QVector3D last = getPoint(numberOfPoints() - 1);
487  QVector3D nexttolast = getPoint(numberOfPoints() - 2);
488  if (last.z() > wlheight && nexttolast.z() > wlheight) {
489  delete_point(numberOfPoints() - 1);
490  } else if (last.z() > wlheight && nexttolast.z() < wlheight) {
491  QVector3D newendpoint(nexttolast.x() + ((last.x() - nexttolast.x())
492  *(wlheight - nexttolast.z()) / (last.z() - nexttolast.z())),
493  nexttolast.y() + ((last.y() - nexttolast.y())
494  *(wlheight - nexttolast.z()) / (last.z() - nexttolast.z())),
495  wlheight);
496  setPoint(numberOfPoints() - 1, newendpoint);
497  } else
498  break;
499  }
500  }
501  Spline::rebuild();
502 }
