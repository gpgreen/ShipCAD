425 static float DisplInterpolate(float x, float x1, float y1, float x2, float y2)
426 {
427  float result;
428 
429  if (fabs(x2 - x1) < 1e-3)
430  result = 0.5 * (y1 + y2);
431  else
432  result = y1 + (y2 - y1) * (x - x1) / (x2 - x1);
433  if (result < y1 || result > y2)
434  result = 0.5 * (y1 + y2);
435  return result;
436 }

466 // used in balance
467 static void CalculateMinMaxData(MinMaxData& mmd, ShipCADModel* owner,
468  Plane& wlplane, float CosTrim, float SinTrim,
469  float CosHeel, float SinHeel)
470 {
471  // zero out min max data
472  mmd.lowest_point = ZERO;
473  mmd.lowest_leak = ZERO;
474  mmd.plane_normal = ZERO;
475  mmd.max_draft = mmd.lowest_z = 0.0;
476  mmd.calculated = false;
477 
478  QVector3D p1 = RotatePointBack(ZERO, CosTrim, SinTrim, CosHeel, SinHeel);
479  QVector3D p2 = RotatePointBack(QVector3D(1.0, 0.0, 0.0),
480  CosTrim, SinTrim, CosHeel, SinHeel);
481  QVector3D p3 = RotatePointBack(QVector3D(1.0, 1.0, 0.0),
482  CosTrim, SinTrim, CosHeel, SinHeel);
483  // the following plane has the right orientation for the given trim
484  // and angle of heel, however the correct distance from the lowest point
485  // on the hull has to be calculated
486  mmd.plane_normal = UnifiedNormal(p1, p2, p3);
487  wlplane = Plane(p1, p2, p3);
488  bool first = true;
489  bool firstleak = true;
490  float min = 0;
491  float max = 0;
492  float distance;
493  SubdivisionSurface* surf = owner->getSurface();
494  for (size_t i=0; i<surf->numberOfLayers(); i++) {
495  if (surf->getLayer(i)->useInHydrostatics()) {
496  SubdivisionLayer* layer = surf->getLayer(i);
497  for (size_t j=0; j<layer->numberOfFaces(); j++) {
498  SubdivisionControlFace* face = layer->getFace(j);
499  for (size_t k=0; k<face->numberOfChildren(); k++) {
500  SubdivisionFace* child = face->getChild(k);
501  for (size_t l=0; l<child->numberOfPoints(); l++) {
502  QVector3D p = child->getPoint(l)->getCoordinate();
503  distance = wlplane.a() * p1.x() + wlplane.b() * p1.y()
504  + wlplane.c() * p1.z() + wlplane.d();
505  if (first) {
506  min = distance;
507  max = min;
508  mmd.lowest_point = p1;
509  mmd.lowest_z = p1.z();
510  } else {
511  if (distance < min) {
512  min = distance;
513  mmd.lowest_point = p1;
514  } else if (distance > max) {
515  max = distance;
516  }
517  if (p1.z() < mmd.lowest_z)
518  mmd.lowest_z = p1.z();
519  }
520  p1.setY(-p1.y());
521  distance = wlplane.a() * p1.x() + wlplane.b() * p1.y()
522  + wlplane.c() * p1.z() + wlplane.d();
523  if (distance < min) {
524  min = distance;
525  mmd.lowest_point = p1;
526  } else if (distance > max)
527  max = distance;
528  // check if this point is a leak point
529  if (fabs(p.y()) > 1e-4 && child->getPoint(l)->isBoundaryVertex()) {
530  p1 = p;
531  if (firstleak) {
532  firstleak = false;
533  mmd.lowest_leak = p1;
534  } else {
535  distance = wlplane.a() * p1.x()
536  + wlplane.b() * p1.y()
537  + wlplane.c() * p1.z() + wlplane.d();
538  float tmp = wlplane.a() * mmd.lowest_leak.x()
539  + wlplane.b() * mmd.lowest_leak.y()
540  + wlplane.c() * mmd.lowest_leak.z()
541  + wlplane.d();
542  if (distance < tmp)
543  mmd.lowest_leak = p1;
544  }
545  p1.setY(-p1.y());
546  distance = wlplane.a() * p1.x()
547  + wlplane.b() * p1.y()
548  + wlplane.c() * p1.z() + wlplane.d();
549  float tmp = wlplane.a() * mmd.lowest_leak.x()
550  + wlplane.b() * mmd.lowest_leak.y()
551  + wlplane.c() * mmd.lowest_leak.z()
552  + wlplane.d();
553  if (distance < tmp)
554  mmd.lowest_leak = p1;
555  }
556  }
557  }
558  }
559  }
560  }
561  if (!firstleak) {
562  // leak points have been found, check if this restricts the max draft
563  distance = wlplane.a() * mmd.lowest_leak.x()
564  + wlplane.b() * mmd.lowest_leak.y()
565  + wlplane.c() * mmd.lowest_leak.z()
566  + wlplane.d();
567  if (distance < max)
568  max = distance;
569  mmd.max_draft = max - min - 1e-4;
570  }

709  QVector3D RotatePoint(QVector3D p)
710  {
711  p.setZ(p.z() - keel.z());
712  return QVector3D(p.x() * CosTrim + p.y() * SinHeel * SinTrim + p.z() * CosHeel * SinTrim,
713  p.y() * CosHeel - p.z() * SinHeel,

740  void ProcessTriangle(QVector3D p1, QVector3D p2, QVector3D p3)
741  {
742  QVector3D volume_moment;
743  // reposition points with respect to the new projected origin
744  p1 -= new_origin;
745  p2 -= new_origin;
746  p3 -= new_origin;
747  QVector3D center = (p1 + p2 + p3) / 3.0f;
748  float volume = QVector3D::dotProduct(p1, QVector3D::crossProduct(p2, p3)) / 6.0f;
749  if (volume != 0) {
750  volume_moment = .75 * volume * center;
751  data.volume += volume;
752  data.center_of_buoyancy += volume_moment;
753  }
754  float ax = 0.5 * ((p1.y() - p2.y()) * (p1.z() + p2.z()) + (p2.y() - p3.y()) * (p2.z() + p3.z())
755  + (p3.y() - p1.y()) * (p3.z() + p1.z()));
756  float ay = 0.5 * ((p1.z() - p2.z()) * (p1.x() + p2.x()) + (p2.z() - p3.z()) * (p2.x() + p3.x())
757  + (p3.z() - p1.z()) * (p3.x() + p1.x()));
758  float az = 0.5 * ((p1.x() - p2.x()) * (p1.y() + p2.y()) + (p2.x() - p3.x()) * (p2.y() + p3.y())
759  + (p3.x() - p1.x()) * (p3.y() + p1.y()));
760  data.wetted_surface += sqrt(ax * ax + ay * ay + az * az);
761  }

763  void run() {
764  QVector3D p, p1, p2, p3;
765  bool submerged = false;
766  float side1, side2, parameter;
767  vector<QVector3D> points;
768  for (size_t i=0; i<hydro_calc->getOwner()->getSurface()->numberOfLayers(); i++) {
769  SubdivisionLayer* layer = hydro_calc->getOwner()->getSurface()->getLayer(i);
770  if (!layer->useInHydrostatics()) continue;
771  for (size_t j=0; j<layer->numberOfFaces(); j++) {
772  SubdivisionControlFace* face = layer->getFace(j);
773  for (size_t k=0; k<face->numberOfChildren(); k++) {
774  SubdivisionFace* child = face->getChild(k);
775  // calculate the portside of the model
776  points.clear();
777  p1 = child->getPoint(child->numberOfPoints()-1)->getCoordinate();
778  // calculate on which side of the waterplane this point is
779  side1 = data.waterline_plane.distance(p1);
780  for (size_t l=0; l<child->numberOfPoints(); l++) {
781  p2 = child->getPoint(l)->getCoordinate();
782  side2 = data.waterline_plane.distance(p2);
783  if ((side1 < -1e-5 && side2 > 1e-5) || (side1 > 1e-5 && side2 < -1e-5)) {
784  // the current linesegment between p1-p2 intersects the waterline plane
785  if (side1 == side2)
786  parameter = .5 * (side1 + side2);
787  else
788  parameter = -side1 / (side2 - side1);
789  p = p1 + parameter * (p2 - p1);
790  CheckSubmergedBody(p, 0);
791  points.push_back(p);
792  }
793  if (side2 <= 1e-5) {
794  if (side2 < data.absolute_draft)
795  data.absolute_draft = side2;
796  // p2 lies also on or under the waterlineplane
797  if (side2 < -1e-5) {
798  // point is submerged, check if the model is making water
799  if (child->getPoint(l)->isBoundaryVertex() && fabs(child->getPoint(l)->getCoordinate().y()) > 1e-4) {
800  if (!hydro_calc->hasError(feMakingWater)) {
801  hydro_calc->addError(feMakingWater);
802  data.leak = child->getPoint(l)->getCoordinate();
803  }
804  }
805  }
806  CheckSubmergedBody(p2, side2);
807  points.push_back(p2);
808  }
809  p1 = p2;
810  side1 = side2;
811  } // end of child face point loop
812 
813  // calculate volume aft of this face
814  if (points.size() > 2)
815  submerged = true;
816  for (size_t l=3; l<=points.size(); l++)
817  ProcessTriangle(points[0], points[l-2], points[l-1]);
818  if (layer->isSymmetric()) {
819  // calculate the starboard side of the model
820  points.clear();
821  p1 = child->getPoint(child->numberOfPoints()-1)->getCoordinate();
822  p1.setY(-p1.y());
823  // calculate on which side of the waterplane this point is
824  side1 = data.waterline_plane.distance(p1);
825  for (size_t l=0; l<child->numberOfPoints(); l++) {
826  p2 = child->getPoint(l)->getCoordinate();
827  p2.setY(-p2.y());
828  side2 = data.waterline_plane.distance(p2);
829  if ((side1 < -1e-5 && side2 > 1e-5) || (side1 > 1e-5 && side2 < -1e-5)) {
830  // the current linesegment between p1-p2 intersects the waterline plane
831  if (side1 == side2)
832  parameter = .5 * (side1 + side2);
833  else
834  parameter = -side1 / (side2 - side1);
835  p = p1 + parameter * (p2 - p1);
836  CheckSubmergedBody(p, 0);
837  points.push_back(p);
838  }
839  if (side2 <= 1e-5) {
840  if (side2 < data.absolute_draft)
841  data.absolute_draft = side2;
842  // p2 lies also on or under the waterlineplane
843  if (side2 < -1e-5) {
844  // point is submerged, check if the model is making water
845  if (child->getPoint(l)->isBoundaryVertex() && fabs(child->getPoint(l)->getCoordinate().y()) > 1e-4) {
846  if (!hydro_calc->hasError(feMakingWater)) {
847  hydro_calc->addError(feMakingWater);
848  data.leak = child->getPoint(l)->getCoordinate();
849  }
850  }
851  }
852  CheckSubmergedBody(p2, side2);
853  points.push_back(p2);
854  }
855  p1 = p2;
856  side1 = side2;
857  } // end of face point loop
858  // calculate volume aft of this face
859  for (size_t l=3; l<=points.size(); l++)
860  ProcessTriangle(points[0], points[l-1], points[l-2]);
861  } // end of symmetric face if
862  } // end of face loop
863  } // end of subdiv control face loop
864  } // end of layers
865 
866  data.absolute_draft = -data.absolute_draft;
867  if (first_point) {
868  // no intersection with the watersurface found, the ship is either
869  // not submerged or totally submerged
870  if (!submerged) {
871  hydro_calc->addError(feNothingSubmerged);
872  data.absolute_draft = 0;
873  }
874  }
875  if (hydro_calc->hasError(feMakingWater)) {
876  data.volume = 0;
877  data.center_of_buoyancy = ZERO;
878  }
879 
880  ProjectSettings& ps = hydro_calc->getOwner()->getProjectSettings();
881 
882  data.displacement = VolumeToDisplacement(data.volume,
883  ps.getWaterDensity(),
884  ps.getAppendageCoefficient(),
885  ps.getUnits());
886  data.length_waterline = data.wl_max.x() - data.wl_min.x();
887  data.beam_waterline = data.wl_max.y() - data.wl_min.y();
888  if (data.volume != 0) {
889  // translate center of buoyancy back to the original origin
890  data.center_of_buoyancy = new_origin + data.center_of_buoyancy / data.volume;
891  data.center_of_buoyancy = RotatePoint(data.center_of_buoyancy);
892  if (data.length_waterline != 0) {
893  data.lcb_perc = 100 * (data.center_of_buoyancy.x() - ps.getMainframeLocation()) / data.length_waterline;
894  }
895  // apply appendage coeff
896  data.volume *= ps.getAppendageCoefficient();
897  }
898  }
