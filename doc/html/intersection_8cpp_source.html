172 void Intersection::draw(Viewport& vp, LineShader* lineshader)
173 {
174  if (vp.getViewportMode() == vmWireFrame) {

176  rebuild();

184  spline->setPenStyle(Qt::DotLine);

186  spline->setPenStyle(Qt::DotLine);
187  if (spline->getPenStyle() == Qt::DotLine)
188  spline->setColor(QColor(0xc0, 0xc0, 0xc0)); // silver
189  spline->setCurvatureColor(_owner->getPreferences().getCurvaturePlotColor());
190  spline->setCurvatureScale(_owner->getVisibility().getCurvatureScale());
191  spline->setShowCurvature(_owner->getVisibility().isShowCurvature() && isShowCurvature());
192  // draw portside
193  bool drawit = _intersection_type == fiButtock || _intersection_type == fiWaterline || _intersection_type == fiDiagonal;
194  if (_intersection_type == fiStation)
195  drawit = vp.getViewportType() != fvBodyplan
196  || _owner->getVisibility().getModelView() == mvBoth
197  || spline->getMax().x() >= _owner->getProjectSettings().getMainframeLocation();
198  if (drawit) {
199  spline->draw(vp, lineshader);
200  }
201  drawit = false;
202  if (_owner->getVisibility().getModelView() == mvBoth)
203  drawit = true;

239  const Plane& intersection_plane)

242  QVector2D ProjectTo2D(const QVector3D& p)
243  {
244  QVector2D result;
245  switch (_intersection_type) {
246  case fiStation:
247  result = QVector2D(p.y(), p.z());

251  break;

261  void operator()(Spline* spline)
262  {
263  bool closed_spline;
264  IntersectionData intersection_data;
265  vector<float> parameters;
266  float spline_area = 0;
267  QVector3D spline_cog = ZERO;
268  QVector2D c = ZERO2;
269  QVector2D momi = c;
270  closed_spline = DistPP3D(spline->getFirstPoint(), spline->getLastPoint()) < 1e-4;
271  if (!closed_spline) {
272  // make it closed
273  spline->add(spline->getFirstPoint());
274  spline->setKnuckle(spline->numberOfPoints()-2, true);
275  }
276  spline->setFragments(500);
277  parameters.push_back(0.0);
278  parameters.push_back(1.0);
279  if (_intersection_type != fiWaterline && spline->intersect_plane(_wl_plane, intersection_data)) {
280  parameters.reserve(intersection_data.number_of_intersections+2);
281  for (size_t i=0; i<intersection_data.parameters.size(); i++)
282  parameters.push_back(intersection_data.parameters[i]);
283  }
284  sort(parameters.begin(), parameters.end());
285  if (parameters.size()) {
286  float t1 = 0;
287  for (size_t i=1; i<parameters.size(); i++) {
288  float t2 = parameters[i];
289  float t = 0.5 * (t1 + t2);
290  QVector3D p = spline->value(t);
291  // check on which side of the plane this point is
292  float side = _wl_plane.distance(p);
293  if (side < 0 || _intersection_type == fiWaterline) {
294  // the point lies at the back of the plane, include this area
295  QVector2D p1 = ZERO2;
296  for (size_t j=0; j<500; j++) {
297  t = t1 + (j/500.0) * (t2-t1);
298  p = spline->value(t);
299  QVector2D p2 = ProjectTo2D(p);
300  if (j > 0) {
301  float delta = 0.5 * (p2.x() + p1.x()) * (p2.y() - p1.y());
302  spline_area += delta;
303  c.setX(c.x() + delta * .25 * (p2.x() + p1.x()));
304  c.setY(c.y() + delta * .5 * (p2.y() + p1.y()));
305  momi.setX(momi.x() + (1.0/12.0) * (p1.y() + p2.y())
306  * (p1.y() * p1.y() + p2.y() * p2.y())
307  * (p2.x() - p1.x()));
308  momi.setY(momi.y() + (1.0/12.0) *
309  (p2.x() * p2.x() * (3 * p2.y() + p1.y())
310  + 2 * p1.x() * p2.x() * (p1.y() + p2.y())
311  + p1.x() * p1.x() * (3 * p1.y() + p2.y())) * (p2.x() - p1.x()));
312  }
313  p1 = p2;
314  }
315  }
316  t1 = t2;
317  }
318  if (spline_area != 0) {
319  c /= spline_area;
320  momi.setX(fabs(momi.x()));
321  momi.setY(fabs(momi.y()));
322  switch (_intersection_type) {
323  case fiStation:
324  spline_cog.setX(-_int_plane.d());
325  spline_cog.setY(c.x());
326  spline_cog.setZ(c.y());
327  _moi->setX(0);
328  _moi->setY(momi.x());
329  _moi->setZ(momi.y());
330  break;
331  case fiButtock:
332  spline_cog.setX(c.x());
333  spline_cog.setY(-_int_plane.d());
334  spline_cog.setZ(c.y());
335  _moi->setX(momi.x());
336  _moi->setY(0);
337  _moi->setZ(momi.y());
338  break;
339  case fiWaterline:
340  spline_cog.setX(c.x());
341  spline_cog.setY(c.y());
342  spline_cog.setZ(-_int_plane.d());
343  _moi->setX(momi.x());
344  _moi->setY(momi.y());
345  _moi->setZ(0);
346  break;
347  default:
348  break;
349  }
350  }
351  }
352  *_area += spline_area;
353  *_cog += spline_area * spline_cog;
354  }

357 void Intersection::calculateArea(const Plane& wlplane, float* area, QVector3D* cog, QVector2D* moment_of_inertia)
358 {
359  *area = 0;
360  *cog = ZERO;
361  *moment_of_inertia = ZERO2;
362  QVector3D tmpmoi;
363  if (!isBuild())
364  rebuild();
365  if (_items.size() > 0) {
366  createStarboardPart(); // this ensures correct winding order
367  CalculateSplineArea calc(area, cog, &tmpmoi, _intersection_type, wlplane, _plane);
368  for_each(_items.begin(), _items.end(), calc);
369  if (*area != 0) {
370  *cog = *cog / *area;
371  if (_intersection_type == fiWaterline) {
372  moment_of_inertia->setX(tmpmoi.x() - cog->y() * cog->y() * *area);
373  moment_of_inertia->setY(tmpmoi.y() - cog->x() * cog->x() * *area);
374  }
375  }
376  }
377  if (_items.size() == 0 || *area == 0) {
378  *cog = ZERO;
379  switch (_intersection_type) {
380  case fiStation:
381  cog->setX(-_plane.d());
382  break;
383  case fiButtock:
384  cog->setY(-_plane.d());
385  break;
386  case fiWaterline:
387  cog->setZ(-_plane.d());
388  break;
389  default:
390  break;
391  }
392  }

395 void Intersection::createStarboardPart()
396 {
397  if (_intersection_type != fiButtock) {
398  // copy all present splines and mirror the y coordinate
399  for (size_t i=_items.size(); i>=1; i--) {
400  Spline* spline = new Spline(*_items.get(i-1));
401  for (size_t j=0; j<spline->numberOfPoints(); j++) {
402  QVector3D p1 = spline->getPoint(j);
403  p1.setY(-p1.y());
404  spline->setPoint(j, p1);
405  }
406  _items.add(spline);
407  }
408  // try to connect the splines
409  JoinSplineSegments(0.05f, false, _items);
410  }
411  // check if the orientation is counterclockwise
412  for (size_t i=_items.size(); i>=1; i--) {
413  Spline* spline = _items.get(i-1);
414  float area = 0;
415  QVector3D p1 = spline->getLastPoint();
416  for (size_t j=0; j<500; j++) {
417  QVector3D p2 = spline->value(j/500.);
418  float delta;
419  switch (_intersection_type) {
420  case fiStation:
421  delta = 0.5 * (p2.y() + p1.y()) * (p2.z() - p1.z());
422  break;
423  case fiButtock:
424  delta = 0.5 * (p2.x() + p1.x()) * (p2.z() - p1.z());
425  break;
426  case fiWaterline:
427  delta = 0.5 * (p2.x() + p1.x()) * (p2.y() - p1.y());
428  break;
429  default:
430  // BUGBUG: better error message
431  throw runtime_error("66");
432  }
433  area += delta;
434  p1 = p2;
435  }
436  if (fabs(area) < 1e-4 && _items.size() > 0) {
437  // either this spline has a very small area (.01m x .01m) or it is an
438  // unconnected straight line. In both cases it may be deleted as long
439  // as at least one valid spline segment remains
440  _items.del(spline);
441  } else if (area < 0) {
442  // spline is defined clockwise, so invert the controlpoints
443  spline->invert_direction();
444  }
445  }
446 }

453 void Intersection::loadBinary(FileBuffer& source)
454 {

458  _intersection_type = static_cast<intersection_type_t>(n);
459  if (_owner->getFileVersion() >= fv191)
460  source.load(_show_curvature);

469  // read number of points for this spline
470  quint32 m;
471  source.load(m);

475  if (_intersection_type == fiStation) {

484  p.setY(-_plane.d());
