63 void Marker::draw(Viewport& vp, LineShader* lineshader)
64 {
65  if (!isVisible())
66  return;
67  if (isSelected())
68  setColor(_owner->getPreferences().getSelectColor());
69  else
70  setColor(_owner->getPreferences().getMarkerColor());
71  setFragments(250);
72 
73  QVector<QVector3D>& vertices = lineshader->getVertexBuffer();
74 
75  QVector3D p3d;
76  if (vp.getViewportType() == fvBodyplan && _owner->getVisibility().getModelView() != mvBoth) {
77 
78  vector<float> params;
79  vector<QVector3D> parray1;
80  vector<QVector3D> parray2;
81 
82  Plane pln(1,0,0,-_owner->getProjectSettings().getMainframeLocation());
83  params.push_back(0);
84  params.push_back(1);
85 
86  IntersectionData output;
87  if (intersect_plane(pln, output)) {
88  params.insert(params.end(), output.parameters.begin(), output.parameters.end());
89  sort(params.begin(), params.end());
90  }
91 
92  for (size_t i=1; i<params.size(); i++) {
93  p3d = value(0.5 * params[i-1] + params[i]);
94  float scale = 1;
95  if (p3d.x() < _owner->getProjectSettings().getMainframeLocation())
96  scale = -1;
97  size_t fragm = floor((params[i] - params[i-1])*getFragments());
98  if (fragm < 10)
99  fragm = 10;
100  for (size_t j=0; j<fragm; j++) {
101  float t = params[i-1] + (params[i] - params[i-1]) * j / (fragm - 1);
102  QVector3D p = value(t);
103  p.setY(p.y() * scale);
104  parray1.push_back(p);
105  }
106  if (showCurvature()) {
107  for (size_t j=0; j<fragm; j++) {
108  float t = params[i-1] + (params[i] - params[i-1]) * j / (fragm - 1);
109  QVector3D normal;
110  float c = curvature(t, normal);
111  normal.setY(normal.y() * scale);
112  QVector3D p2 = parray1[j] - (c * 2 * getCurvatureScale() * normal);
113  parray2.push_back(p2);
114  }
115  for (size_t j=1; j<=fragm; j++) {
116  if ((j % 4) == 0 || j == 1 || j == fragm) {
117  vertices << parray1[j-1];
118  vertices << parray2[j-1];
119  }
120  }
121  for (size_t j=1; j<parray2.size(); j++) {
122  vertices << parray2[j-1];
123  vertices << parray2[j];
124  }
125  glLineWidth(1);
126  lineshader->renderLines(vertices, getCurvatureColor());
127  } else { // end show curvature
128  for (size_t j=1; j<parray1.size(); j++) {
129  vertices << parray1[j-1];
130  vertices << parray1[j];
131  }
132  glLineWidth(1);
133  lineshader->renderLines(vertices, getColor());
134  }
135  }
136  vertices.clear();
137  for (size_t i=0; i<numberOfPoints(); i++) {
138  p3d = getPoint(i);
139  if (p3d.x() < _owner->getProjectSettings().getMainframeLocation())
140  p3d.setY(-p3d.y());
141  vertices << p3d;
142  }
143  glPointSize(3);
144  lineshader->renderPoints(vertices, getColor());
145  } else { // end bodyplan
146  Spline::draw(vp, lineshader);
147  for (size_t i=0; i<numberOfPoints(); i++) {
148  vertices << getPoint(i);
149  }
150  glPointSize(3);
151  lineshader->renderPoints(vertices, getColor());
152  }
153 }

189 void Marker::loadFromText(ShipCADModel* model, QTextStream& file, MarkerVector& markers)
190 {
191  size_t lineno = 0;
192  bool unexpected = false;
193  Marker* marker = nullptr;
194  while (!unexpected && !file.atEnd()) {
195  QString line = file.readLine();
196  lineno++;
197  // if an empty line, complete the marker
198  // and get ready for more
199  if (line.size() == 0 && marker != nullptr) {
200  if (marker->numberOfPoints() > 1) {
201  markers.add(marker);
202  } else {
203  delete marker;
204  }
205  marker = nullptr;
206  continue;
207  }
208  if (line == "EOF")
209  break;
210  // split the line
211  QStringList fields = line.split(" ");
212  if (fields.size() != 3) {
213  // try tabs
214  fields = line.split("\t");
215  if (fields.size() != 3) {
216  unexpected = true;
217  continue;
218  }
219  }
220  // we have 3 pieces, turn it into a point
221  bool ok;
222  float x = fields[0].toFloat(&ok);
223  if (!ok) {
224  unexpected = true;
225  continue;
226  }
227  float y = fields[1].toFloat(&ok);
228  if (!ok) {
229  unexpected = true;
230  continue;
231  }
232  float z = fields[2].toFloat(&ok);
233  if (!ok) {
234  unexpected = true;
235  continue;
236  }
237  // add the point to the marker
238  QVector3D pt(x, y, z);
239  if (marker == nullptr) {
240  marker = Marker::construct(model);
241  }
242  marker->add(pt);
243  }
244  // complete the last marker
245  if (!unexpected && marker != nullptr) {
246  if (marker->numberOfPoints() > 1) {
247  markers.add(marker);
248  } else
249  delete marker;
250  }
251  // got something wrong, bail
252  if (unexpected) {
253  throw ParseError(lineno, "");
254  }
255 }
