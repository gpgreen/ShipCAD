147 void OpenGLWindow::renderNow()
148 {
149  if (!isExposed())
150  return;
151 
152  m_update_pending = false;
153 
154  bool needsInitialize = false;
155 
156  if (!m_context) {
157  m_context = new QOpenGLContext(this);
158  m_context->setFormat(requestedFormat());
159  m_context->create();
160 
161  needsInitialize = true;
162  }
163 
164  m_context->makeCurrent(this);
165 
166  if (needsInitialize) {
167  initializeOpenGLFunctions();
168  initialize();
169  }
170 
171  render();
172 
173  m_context->swapBuffers(this);
174 
175  if (m_animating)
176  renderLater();
177 }
