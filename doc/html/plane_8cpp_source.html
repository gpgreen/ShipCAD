58 Plane::Plane(const QVector3D& p1, const QVector3D& p2, const QVector3D& p3)
59 {
60  // calculate normal
61  QVector3D n;
62  n.setX((p2.y() - p1.y()) * (p3.z() - p1.z()) - (p2.z() - p1.z()) * (p3.y() - p1.y()));
63  n.setY((p2.z() - p1.z()) * (p3.x() - p1.x()) - (p2.x() - p1.x()) * (p3.z() - p1.z()));
64  n.setZ((p2.x() - p1.x()) * (p3.y() - p1.y()) - (p2.y() - p1.y()) * (p3.x() - p1.x()));
65  n.normalize();
66  _vars[0] = n.x() / n.length();
67  _vars[1] = n.y() / n.length();
68  _vars[2] = n.z() / n.length();
69  _vars[3] = -p1.x() * _vars[0] - p1.y() * _vars[1] - p1.z() * _vars[2];
70 }

124 QVector3D Plane::projectPointOnPlane(const QVector3D& p) const
125 {
126  float q = _vars[0] * _vars[0] + _vars[1] * _vars[1] + _vars[2] * _vars[2];
127  if (q != 0) {
128  float r = (_vars[0] * p.x() + _vars[1] * p.y() + _vars[2] * p.z() + _vars[3]) / q;
129  return QVector3D(p.x() - _vars[0] * r,

136 bool Plane::intersectLine(const QVector3D& pt, const QVector3D& n, bool& coplanar, QVector3D& intpt) const
137 {
