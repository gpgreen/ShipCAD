84 // predicate class to return true if pt1 is z ordered less than pt2

88  ShipCAD::SubdivisionControlPoint* pt2)
89  {
90  QVector3D p1 = pt1->getCoordinate();
91  QVector3D p2 = pt2->getCoordinate();
92  if (p1.z() < p2.z())
93  return true;
94  if (abs(p1.z() - p2.z()) < 1E-6) {
95  if (p1.x() < p2.x())
96  return true;
97  if (abs(p1.x() - p2.x()) < 1E-6) {
98  if (p1.y() < p2.y())
99  return true;
100  }
101  }
102  return false;
103  }
104 };
