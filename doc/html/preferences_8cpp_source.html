129 void Preferences::readSettings()
130 {
131  // the colors
132  map<int, ColorChanger> colors;
133  getColorDialogMap(colors);
134  QSettings settings;
135  int size = settings.beginReadArray("preferences/color");
136  if (size > 0) {
137  for (int i=0; i<size && i<colors.size(); ++i) {
138  settings.setArrayIndex(i);
139  int red = settings.value("red").toInt();
140  int green = settings.value("green").toInt();
141  int blue = settings.value("blue").toInt();
142  map<int, ColorChanger>::iterator j = colors.find(i);
143  if (j == colors.end())
144  continue;
145  ColorChanger& changer = (*j).second;
146  *(changer.setColor) = QColor(red, green, blue);
147  }
148  }
149  settings.endArray();
150  // the rest
151  if (settings.contains("preferences/pointsize"))
152  _point_size = settings.value("preferences/pointsize").toInt();
153  if (settings.contains("preferences/maxundomemory"))
154  _max_undo_memory = settings.value("preferences/maxundomemory").toInt();
155 
156 }

158 void Preferences::saveSettings() const
159 {
160  // colors
161  map<int, ColorChanger> colors;
162  const_cast<Preferences*>(this)->getColorDialogMap(colors);
163  QSettings settings;
164  settings.beginWriteArray("preferences/color");
165  for (int i=0; i<colors.size(); ++i) {
166  settings.setArrayIndex(i);
167  map<int, ColorChanger>::iterator j = colors.find(i);
168  if (j == colors.end())
169  continue;
170  ColorChanger& changer = (*j).second;
171  settings.setValue("red", (*changer.setColor).red());
172  settings.setValue("green", (*changer.setColor).green());
173  settings.setValue("blue", (*changer.setColor).blue());
174  }
175  settings.endArray();
176  // the rest
177  settings.setValue("preferences/pointsize", _point_size);
178  settings.setValue("preferences/maxundomemory", static_cast<int>(_max_undo_memory));
179 }
