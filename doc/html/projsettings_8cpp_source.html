301 void ProjectSettings::changeUnitsInSettingsOnly(unit_type_t unit)
302 {
303  double unit_conversion_factor;
304  float weight_factor;
305 
306  if (unit == _units)
307  return;
308  _units = unit;
309  if (_units == fuImperial) {
310  unit_conversion_factor = 1 / 0.3048;
311  weight_factor = kWeightConversionFactor;
312  } else {
313  unit_conversion_factor = 0.3048;
314  weight_factor = 1 / kWeightConversionFactor;
315  }
316  _water_density *= weight_factor;
317  setLength(abs(getLength() * unit_conversion_factor));
318  setBeam(abs(getBeam() * unit_conversion_factor));
319  setDraft(abs(getDraft() * unit_conversion_factor));
320  if (!useDefaultMainframeLocation())
321  setMainframeLocation(getMainframeLocation() * unit_conversion_factor);
322 }

324 bool ProjectSettings::setUnits(unit_type_t unit)
325 {
326  double unit_conversion_factor;
327  float weight_factor;
328  float thickness_factor;
329 
330  if (unit == _units)
331  return false;
332  _units = unit;
333  if (_units == fuImperial) {
334  unit_conversion_factor = 1 / 0.3048;
335  weight_factor = kWeightConversionFactor;
336  thickness_factor = 1 / 25.4f;
337  } else {
338  unit_conversion_factor = 0.3048;
339  weight_factor = 1 / kWeightConversionFactor;
340  thickness_factor = 25.4f;
341  }
342  _water_density *= weight_factor;
343  for (size_t i=0; i<_owner->getSurface()->numberOfLayers(); i++) {
344  SubdivisionLayer *layer = _owner->getSurface()->getLayer(i);
345  layer->setMaterialDensity(layer->getMaterialDensity() * weight_factor);
346  layer->setThickness(layer->getThickness() * thickness_factor);
347  }
348  QVector3D scale_vector(unit_conversion_factor,
349  unit_conversion_factor,
350  unit_conversion_factor);
351  _owner->scaleModel(scale_vector, true, true); // scale and redraw
352  return true;
353 }
