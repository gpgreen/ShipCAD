91 void ShipCADModel::buildValidFrameTable(SplineVector& dest, bool close_at_deck)
92 {
93  float min = 0;
94  IntersectionVector& stations = getStations();
95  for (size_t i=0; i<stations.size(); ++i) {
96  Intersection* intersection = stations.get(i);
97  if (!intersection->isBuild())
98  intersection->rebuild();
99  SplineVector tmp(false);
100  SplineVector& splines = intersection->getSplines();
101  for (size_t j=0; j<splines.size(); j++) {
102  Spline* spline = new Spline(*(splines.get(j)));
103  // quick check to determine if the frame runs from bottom to top
104  if (spline->value(0.0).z() > spline->value(1.0).z()) {
105  // if not reverse the points
106  spline->invert_direction();
107  }
108  tmp.add(spline);
109  }
110  // take all segments and join into one
111  if (tmp.size() > 1)
112  JoinSplineSegments(0.01, true, tmp);
113  for (size_t j=0; j<tmp.size(); j++) {
114  Spline* spline = tmp.get(j);
115  if (close_at_deck) {
116  if (spline->getLastPoint().y() != 0.0) {
117  QVector3D p = spline->getLastPoint();
118  p.setY(0.0);
119  spline->add(p);
120  spline->setKnuckle(spline->numberOfPoints() - 2, true);
121  }
122  }
123  dest.add(spline);
124  if (i == 0)
125  min = spline->getMin().z();
126  else if (spline->getMin().z() < min)
127  min = spline->getMin().z();
128  }
129  }
130  // now shift all stations up or down so that the lowest point
131  // of all stations is on the baseline z=0.0
132  if (min != 0.0) {
133  for (size_t i=0; i<dest.size(); i++) {
134  Spline* spline = dest.get(i);
135  for (size_t j=0; j<spline->numberOfPoints(); j++) {
136  QVector3D p = spline->getPoint(j);
