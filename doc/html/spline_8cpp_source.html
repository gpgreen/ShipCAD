114 void Spline::rebuild()
115 {
116  // attempt to eliminate double points
117  _total_length = 0;
118  for (size_t i=1; i<_nopoints; ++i) {
119  _total_length += sqrt(_points[i-1].distanceToPoint(_points[i]));
120  }
121  vector<QVector3D> u;
122  QVector3D un, qn;
123  float p;
124 
125  if (_nopoints > 1) {
126  _derivatives.clear();
127  _derivatives.reserve(_nopoints);
128  _parameters.clear();
129  _parameters.reserve(_nopoints);
130  u.reserve(_nopoints);
131 
132  if (fabs(_total_length) < 1E-5) {
133  // zero arc length, use uniform parameterization
134  for (size_t i=0; i<_nopoints; ++i) {
135  _parameters.push_back(i / static_cast<float>(_nopoints-1));
136  }
137  }
138  else {
139  _parameters.push_back(0);
140  float length = 0;
141  for (size_t i=1; i<_nopoints-1; ++i) {
142  length += sqrt(_points[i-1].distanceToPoint(_points[i]));
143  _parameters.push_back(length / _total_length);
144  }
145  _parameters.push_back(1.0);
146  }
147 
148  _derivatives.push_back(ZERO);
149  u.push_back(_derivatives[0]);
150 
151  for (size_t i=1; i<_nopoints-1; ++i) {
152  if (_knuckles[i]) {
153  u.push_back(ZERO);
154  _derivatives.push_back(ZERO);
155  }
156  else {
157  if (fabs(_parameters[i+1] - _parameters[i-1]) < 1e-5
158  || fabs(_parameters[i] - _parameters[i-1]) < 1e-5
159  || fabs(_parameters[i+1] - _parameters[i]) < 1e-5) {
160  u.push_back(ZERO);
161  _derivatives.push_back(ZERO);
162  }
163  else {
164  float sig = (_parameters[i] - _parameters[i-1])
165  / (_parameters[i+1] - _parameters[i-1]);
166  // first x-value
167  p = sig * _derivatives[i-1].x() + 2.0;
168  QVector3D p1;
169  QVector3D p2;
170  p1.setX((sig - 1.0) / p);
171  p2.setX( (6.0 * ((_points[i+1].x() - _points[i].x())
172  / (_parameters[i+1] - _parameters[i])
173  - (_points[i].x() - _points[i-1].x())
174  / (_parameters[i] - _parameters[i-1]))
175  / (_parameters[i+1] - _parameters[i-1])
176  - sig * u[i-1].x()) / p);
177  // then y-value
178  p = sig * _derivatives[i-1].y() + 2.0;
179  p1.setY((sig - 1.0) / p);
180  p2.setY( (6.0 * ((_points[i+1].y() - _points[i].y())
181  / (_parameters[i+1] - _parameters[i])
182  - (_points[i].y() - _points[i-1].y())
183  / (_parameters[i] - _parameters[i-1]))
184  / (_parameters[i+1] - _parameters[i-1])
185  - sig * u[i-1].y()) / p);
186  // then z-value
187  p = sig * _derivatives[i-1].z() + 2.0;
188  p1.setZ((sig - 1.0) / p);
189  p2.setZ( (6.0 * ((_points[i+1].z() - _points[i].z())
190  / (_parameters[i+1] - _parameters[i])
191  - (_points[i].z() - _points[i-1].z())
192  / (_parameters[i] - _parameters[i-1]))
193  / (_parameters[i+1] - _parameters[i-1])
194  - sig * u[i-1].z()) / p);
195  _derivatives.push_back(p1);
196  u.push_back(p2);
197  }
198  }
199  }
200 
201  // BUGBUG: these are always 0, how does it work
202  qn = ZERO;
203  un = qn;
204 
205  QVector3D p3;
206  p3.setX(un.x() - qn.x() * u[_nopoints-2].x()
207  / (qn.x() * _derivatives[_nopoints-2].x() + 1.0));
208  p3.setY(un.y() - qn.y() * u[_nopoints-2].y()
209  / (qn.y() * _derivatives[_nopoints-2].y() + 1.0));
210  p3.setZ(un.z() - qn.z() * u[_nopoints-2].z()
211  / (qn.z() * _derivatives[_nopoints-2].z() + 1.0));
212  _derivatives.push_back(p3);
213 
214  // back substitution
215  for (size_t k=_nopoints-1; k>=1; --k) {
216  _derivatives[k-1] = _derivatives[k-1] * _derivatives[k] + u[k-1];
217  }
218  } // end _nopoints > 1
219  _build = true;
220  // determine min/max values
221  if (_nopoints > 0) {
222  _min = _points[0];
223  _max = _min;
224  for (size_t i=1; i<_nopoints; ++i) {
225  MinMax(_points[i], _min, _max);
226  }
227  }
228 }

230 QVector3D Spline::second_derive(float parameter) const
231 {
232  QVector3D result = ZERO;
233  if (_nopoints < 2)
234  return result;
235  if (!_build)
236  const_cast<Spline*>(this)->rebuild();
237  if (_nopoints < 2)
238  return result;
239 
240  size_t lo, hi, k;
241 
242  if (_nopoints == 2) {
243  lo = 0;
244  hi = 1;
245  }
246  else {
247  lo = 0;
248  hi = _nopoints - 1;
249  do {
250  k = (lo + hi) / 2;
251  if (_parameters[k] < parameter)
252  lo = k;
253  else
254  hi = k;
255  } while (hi - lo > 1);
256  }
257  float frac;
258  if (_parameters[hi] - _parameters[lo] <= 0.0)
259  frac = 0.5;
260  else
261  frac = (parameter - _parameters[lo]) / (_parameters[hi] - _parameters[lo]);
262  result = _derivatives[lo] + (frac * (_derivatives[hi] - _derivatives[lo]));
263  return result;
264 }

266 float Spline::weight(size_t index, float total_length)
267 {
268  float result;
269  float length, dist;
270 
271  if (index == 0 || (index == _nopoints - 1) || _knuckles[index]) {
272  result = 1E10;
273  }
274  else {
275  QVector3D p1 = _points[index-1];
276  QVector3D p2 = _points[index];
277  QVector3D p3 = _points[index+1];
278  length = (p3 - p1).length();
279  if (length < 1E-5) {
280  result = 0.0;
281  }
282  else {
283  dist = DistancepointToLine(p2,p1,p3);
284  if (dist < 1E-2) {
285  if (length*length/total_length > 0.01)
286  result = 1E10;
287  else
288  result = 1E8 * dist * dist * length;
289  }
290  else
291  result = 1E8 * dist * dist * length;
292  }
293  }
294  return result;
295 }

319 bool Spline::simplify(float criterium)
320 {
321  if (!isBuild())
322  rebuild();
323 
324  if (_nopoints < 3)
325  return true;
326 
327  vector<float> weights;
328  weights.reserve(_nopoints);
329 
330  float total_length = _total_length * _total_length;
331  if (total_length == 0)
332  return false;
333 
334  for (size_t i=0; i<_nopoints; ++i)
335  weights.push_back(weight(i, total_length)/total_length);
336 
337  // BUGBUG: using an iterator, but logic erases
338  // element from vector which will invalidate the iterator
339  // need to fix....
340  vector<float>::iterator index;
341  do {
342  index = find_next_point(weights);
343  vector<float>::iterator last = weights.end();
344  last--;
345  if (index == weights.end() || _nopoints < 3 || index == last)
346  break;
347  if (*index < criterium) {
348  // use int as we want to test for negative below
349  int j = index - weights.begin();
350  weights.erase(index);
351  _points.erase(_points.begin()+j);
352  _knuckles.erase(_knuckles.begin()+j);
353  _nopoints--;
354  if (j-1 >= 0 && j-1 < static_cast<int>(_nopoints))
355  weights[j-1] = weight(j-1, total_length)/total_length;
356  if (j >= 0 && j < static_cast<int>(_nopoints))
357  weights[j] = weight(j, total_length)/total_length;
358  if (j+1 >= 0 && j+1 < static_cast<int>(_nopoints))
359  weights[j+1] = weight(j+1, total_length)/total_length;
360  }
361  else
362  break;
363  }
364  while (index != weights.end());
365 
366  setBuild(false);
367 
368  return true;
369 }

379 float Spline::coord_length(float t1, float t2) const
380 {
381  float result = 0.0;

384  if (!_build)

399 float Spline::chord_length_approximation(float percentage) const
400 {
401  float result, length, total_length;
402  float t1, t2;
403  t1 = 0;
404  t2 = 1.0;
405  float parameter = 1.0;
406  int counter = 0;
407  float l1 = 0;
408  float l2 = 0;
409  float desired_length = 0;
410  if (percentage < 0)
411  result = 0;
412  else if (percentage > 1)
413  result = 1;
414  else {
415  do {
416  length = coord_length(0, parameter);
417  if (counter == 0) {
418  l1 = 0;
419  l2 = length;
420  total_length = length;
421  desired_length = percentage*total_length;
422  parameter = percentage;
423  }
424  else {
425  if (length > desired_length) {
426  t2 = parameter;
427  l2 = length;
428  }
429  else {
430  t1 = parameter;
431  l1 = length;
432  }
433  parameter = t1 + ((desired_length - l1) / (l2-l1)) * (t2 - t1);
434  if (parameter < 0)
435  parameter = 0;
436  if (parameter > 1)
437  parameter = 1;
438  }
439  counter++;
440  }
441  while (counter <= 75 && fabs(length - desired_length) >= 1E-3);
442  result = parameter;
443  }
444  return result;
445 }

447 float Spline::curvature(float parameter, QVector3D& normal) const
448 {
449  float result;
450 
451  QVector3D vel1 = first_derive(parameter);
452  QVector3D acc = second_derive(parameter);
453 
454  QVector3D crossproduct = QVector3D::crossProduct(acc, vel1);
455  float l = crossproduct.length();
456  if (l == 0)
457  l = 0.00001f;
458  float vdota = (vel1.x()*acc.x()) + (vel1.y()*acc.y()) + (vel1.z()*acc.z());
459  float vdotv = (vel1.x()*vel1.x()) + (vel1.y()*vel1.y()) + (vel1.z()*vel1.z());
460  if (vdotv < 0)
461  result = 0;
462  else
463  result = l/pow(vdotv, static_cast<float>(1.5));
464  normal = (vdotv * acc) - (vdota * vel1);
465  normal.normalize();
466  return result;
467 }

657  // if the new point is very close to the last point, don't add it
658  // this will be the case if intersection is at the end point of a line segment
659  if (output.number_of_intersections > 0 && fabs(parameter - output.parameters.back()) < 1E-6)
660  return;
661  output.number_of_intersections++;
662  output.points.push_back(p);
663  output.parameters.push_back(parameter);
664 }

666 bool Spline::intersect_plane(const Plane& plane, IntersectionData& output) const
667 {
668  output.number_of_intersections = 0;
669  float t1 = 0;
670  QVector3D p1 = value(t1);
671  float s1 = plane.distance(p1);
672  if (fabs(s1) < 1E-6)
673  add_to_output(p1, t1, output);
674  for (size_t i=1; i<=_fragments; ++i) {
675  float t2 = i / static_cast<float>(_fragments);
676  QVector3D p2 = value(t2);
677  float s2 = plane.distance(p2);
678  if (fabs(s2) < 1E-6)
679  add_to_output(p2, t2, output);
680  if (s1 < 0 && s2 > 0) {
681  // intersection found
682  float t = s1 / (s2 - s1);
683  t = t1 - t * (t2 - t1);
684  add_to_output(value(t), t, output);
685  } else if (s2 < 0 && s1 > 0) {
686  // intersection found
687  float t = -s1 / (s2 - s1);
688  t = t1 + t * (t2 - t1);
689  add_to_output(value(t), t, output);
690  }
691  p1 = p2;
692  s1 = s2;
693  t1 = t2;
694  }
695  return output.number_of_intersections > 0;
696 }

699 {

806 QVector3D Spline::value(float parameter) const
807 {
808  QVector3D result;
809  if (_nopoints < 2)
810  return result;
811  if (!_build)
812  const_cast<Spline*>(this)->rebuild();
813  if (_nopoints < 2)
814  return result;
815 
816  size_t lo, hi;
817  if (_nopoints == 2) {
818  lo = 0;
819  hi = 1;
820  }
821  else {
822  lo = 0;
823  hi = _nopoints - 1;
824  do {
825  size_t k = (lo + hi) / 2;
826  if (_parameters[k] < parameter)
827  lo = k;
828  else
829  hi = k;
830  }
831  while (hi - lo > 1);
832  }
833  float h, a, b;
834  h = _parameters[hi] - _parameters[lo];
835  if (fabs(h) < 1E-6) {
836  result = _points[hi];
837  }
838  else {
839  a = (_parameters[hi] - parameter) / h;
840  //b = (parameter - _parameters[lo]) / h;
841  b = 1 - a;
842  result = a * _points[lo] + b * _points[hi] + ((a * a * a - a) * _derivatives[lo]
843  + (b * b * b - b) * _derivatives[hi])
844  * (h * h) / 6.0;
845  }
846  return result;
847 }
