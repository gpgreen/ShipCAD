145 void SubdivisionControlCurve::deleteEdge(SubdivisionControlEdge* edge)
146 {
147  bool delcurve = false;
148  SubdivisionControlPoint* p1, *p2;
149  SubdivisionEdge* anedge;
150 
151  size_t i = 2;
152  while (i <= _points.size()) {
153  if ((_points[i-2] == edge->startPoint() && _points[i-1] == edge->endPoint())
154  || (_points[i-2] == edge->endPoint() && _points[i-1] == edge->startPoint())) {
155  // remove references to this curve from control edges
156  for (size_t j=2; j<=_points.size(); ++j) {
157  anedge = _owner->edgeExists(_points[j-2], _points[j-1]);
158  if (anedge != 0)
159  anedge->setCurve(0);
160  }
161  // remove references from subdivided edges
162  for (size_t j=2; j<=_div_points.size(); ++j) {
163  anedge = _owner->edgeExists(_div_points[j-2], _div_points[j-1]);
164  if (anedge != 0)
165  anedge->setCurve(0);
166  }
167  _div_points.clear();
168 
169  if (i - 2 > 0) {
170  // build first new curve
171  SubdivisionControlCurve* newcurve = SubdivisionControlCurve::construct(_owner);
172  _owner->addControlCurve(newcurve);
173  p1 = 0;
174  for (size_t j=0; j<=i-2; ++j) {
175  p2 = _points[j];
176  newcurve->addPoint(p2);
177  if (j > 0) {
178  anedge = _owner->edgeExists(p1, p2);
179  if (anedge != 0)
180  anedge->setCurve(newcurve);
181  }
182  p1 = p2;
183  }
184  newcurve->setSelected(isSelected());
185  }
186  if (i - 1 < _points.size() - 1) {
187  // build second new curve
188  SubdivisionControlCurve* newcurve = SubdivisionControlCurve::construct(_owner);
189  _owner->addControlCurve(newcurve);
190  p1 = 0;
191  for (size_t j=i-1; j<=_points.size()-1; ++j) {
192  p2 = _points[j];
193  newcurve->addPoint(p2);
194  if (j > i - 1) {
195  anedge = _owner->edgeExists(p1, p2);
196  if (anedge != 0)
197  anedge->setCurve(newcurve);
198  }
199  p1 = p2;
200  }
201  newcurve->setSelected(isSelected());
202  }
203  delcurve = true;
204  break;
205  }
206  ++i;
207  } // end while
208  if (delcurve) {
209  _owner->deleteControlCurve(this);
210  }
211 }

213 void SubdivisionControlCurve::draw(Viewport &vp, LineShader* lineshader)
214 {
215  if (numberOfControlPoints() <= 1)
216  return;
217  bool sel = isSelected();
218  Spline* curve = getSpline();
219  curve->setColor(_owner->getControlCurveColor());
220  curve->setShowCurvature(sel && _owner->showCurvature());
221  if (curve->showCurvature())
222  curve->setFragments(600);
223  else
224  curve->setFragments(250);
225  curve->setCurvatureColor(_owner->getCurvatureColor());
226  curve->setCurvatureScale(_owner->getCurvatureScale());
227 
228  QVector<QVector3D>& vertices = lineshader->getVertexBuffer();
229 
230  if (!_owner->showControlNet() && sel) {
231  // draw controlpoints and edges
232  for (size_t i=1; i<_points.size(); ++i) {
233  SubdivisionPoint* p1 = _points[i-1];
234  SubdivisionPoint* p2 = _points[i];
235  SubdivisionControlEdge* edge = _owner->controlEdgeExists(p1, p2);
236  if (edge != 0)
237  edge->draw(vp, lineshader);
238  if (i == 1)
239  vertices << p1->getCoordinate();
240  vertices << p2->getCoordinate();
241  }
242  lineshader->renderPoints(vertices, getColor());
243  }
244  vertices.clear();
245  if (vp.getViewportType() == fvBodyplan && !_owner->drawMirror()) {
246 
247  vector<QVector3D> parray1;
248  vector<QVector3D> parray2;
249 
250  // draw mainframe location
251  Plane mfl(1, 0, 0, -_owner->getMainframeLocation());
252  IntersectionData output;
253  vector<float> parameters;
254  parameters.push_back(0);
255  parameters.push_back(1.0);
256  if (curve->intersect_plane(mfl, output)) {
257  parameters.insert(parameters.end(), output.parameters.begin(), output.parameters.end());
258  sort(parameters.begin(), parameters.end());
259  }
260  QVector<QVector3D> curvelines;
261  for (size_t i=2; i<=parameters.size(); ++i) {
262  QVector3D p3d = curve->value(0.5*(parameters[i-2] + parameters[i-1]));
263  int scale;
264  if (p3d.x() < _owner->getMainframeLocation())
265  scale = -1;
266  else
267  scale = 1;
268  size_t fragm = (parameters[i-1] - parameters[i-2]) * curve->getFragments();
269  if (fragm < 10)
270  fragm = 10;
271  for (size_t j=0; j<fragm; j++) {
272  float t = parameters[i-1] + (parameters[i] - parameters[i-1]) * j / (fragm - 1);
273  QVector3D p = curve->value(t);
274  p.setY(p.y() * scale);
275  parray1.push_back(p);
276  }
277  if (curve->showCurvature()) {
278  for (size_t j=0; j<fragm; ++j) {
279  float t = parameters[i-2] + (parameters[i-1] - parameters[i-2]) * j / (fragm - 1);
280  QVector3D normal;
281  float c = curve->curvature(t, normal);
282  normal.setY(normal.y() * scale);
283  QVector3D p3d2 = parray1[j] - (c * 2 * curve->getCurvatureScale() * normal);
284  parray2.push_back(p3d2);
285  }
286  for (size_t j=1; j<=fragm; j++) {
287  if ((j % 4) == 0 || j == 1 || j == fragm) {
288  // draw normal lines
289  vertices << parray1[j-1];
290  vertices << parray2[j-1];
291  }
292  }
293  for (size_t j=1; j<parray2.size(); j++) {
294  vertices << parray2[j-1];
295  vertices << parray2[j];
296  }
297  glLineWidth(1);
298  lineshader->renderLines(vertices, curve->getCurvatureColor());
299  }
300  else {
301  for (size_t j=1; j<=parray1.size(); ++j) {
302  vertices << parray1[j-1];
303  vertices << parray1[j];
304  }
305  glLineWidth(1);
306  lineshader->renderLines(vertices, getColor());
307  }
308  }
309  }
310  else {
311  curve->draw(vp, lineshader);
312  }
313  if (_owner->drawMirror()) {
314  // draw reversed curve
315  for (size_t i=1; i<=curve->numberOfPoints(); ++i) {
316  QVector3D p3d = curve->getPoint(i-1);
317  p3d.setY(-p3d.y());
318  curve->setPoint(i-1, p3d);
319  }
320  curve->draw(vp, lineshader);
321  for (size_t i=1; i<=curve->numberOfPoints(); ++i) {
322  QVector3D p3d = curve->getPoint(i-1);
323  p3d.setY(-p3d.y());
324  curve->setPoint(i-1, p3d);
325  }
326  }
327 }

357 void SubdivisionControlCurve::loadBinary(FileBuffer &source)
358 {
359  quint32 n, ind;
360  source.load(n);
361  SubdivisionControlPoint* p1 = 0;
362  for (size_t i=0; i<n; ++i) {
363  source.load(ind);

365  _points.push_back(p2);
366  if (i > 0) {

371  p1 = p2;
