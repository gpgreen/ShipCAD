79 // algorithm from http://geomalgorithms.com/a07-_distance.html#Distance-between-Segments-and-Rays
80 float SubdivisionEdge::distanceToEdge(const QVector3D& pt, const QVector3D& dir) const
81 {
82  QVector3D l2p1 = pt + 1000*dir;
83  QVector3D u = _points[1]->getCoordinate() - _points[0]->getCoordinate();
84  QVector3D v = l2p1 - pt;
85  QVector3D w = _points[0]->getCoordinate() - pt;
86  float a = QVector3D::dotProduct(u, u); // always >= 0
87  float b = QVector3D::dotProduct(u, v);
88  float c = QVector3D::dotProduct(v, v); // always >= 0
89  float d = QVector3D::dotProduct(u, w);
90  float e = QVector3D::dotProduct(v, w);
91  float D = a * c - b * b; // always >= 0
92  float sc, sN, sD = D; // sc = sN / sD, default sD = D >= 0
93  float tc, tN, tD = D; // tc = tN / tD, default tD = D >= 0
94 
95  // compute the line parameters of the two closest points
96  if (D < 1E-5) { // the lines are almost parallel
97  sN = 0.0; // force using point P0 on this edge
98  sD = 1.0; // to prevent possible division by 0 later
99  tN = e;
100  tD = c;
101  }
102  else { // get the closest points on the infinite lines
103  sN = (b * e - c * d);
104  tN = (a * e - b * d);
105  if (sN < 0.0) { // sc < 0 => the s=0 edge is visible
106  sN = 0.0;
107  tN = e;
108  tD = c;
109  } else if (sN > sD) { // sc > 1 => the s=1 edge is visible
110  sN = sD;
111  tN = e + b;
112  tD = c;
113  }
114  }
115  if (tN < 0.0) { // tc < 0 => the t=0 edge is visible
116  tN = 0.0;
117  // recompute sc for this edge
118  if (-d < 0.0)
119  sN = 0.0;
120  else if (-d > a)
121  sN = sD;
122  else {
123  sN = -d;
124  sD = a;
125  }
126  }
127  else if (tN > tD) { // tc > 1 => the t=e edge is visible
128  tN = tD;
129  // recompute sc for this edge
130  if ((-d + b) < 0.0)
131  sN = 0;
132  else if ((-d + b) > a)
133  sN = sD;
134  else {
135  sN = (-d + b);
136  sD = a;
137  }
138  }
139  // do the division to get sc and tc
140  sc = (abs(sN) < 1E-5 ? 0.0 : sN / sD);
141  tc = (abs(tN) < 1E-5 ? 0.0 : tN / tD);
142 
143  // get the difference of the two closest points
144  QVector3D dP = w + (sc * u) - (tc * v); // = S1(sc) - S2(sc)
145 
146  return dP.length();
147 }

168 void SubdivisionEdge::setCrease(bool val)
169 {
170  if (_faces.size() == 1)
171  val = true; // boundary edges must always be crease edges
172  if (val == _crease)
173  return;
174  _crease = val;
175  int n = 0;
176  SubdivisionPoint* startPoint = _points[0];
177  SubdivisionPoint* endPoint = _points[1];
178  for (size_t i = 0; i<startPoint->numberOfEdges(); ++i) {
179  SubdivisionEdge* edge = startPoint->getEdge(i);
180  if (edge->isCrease())
181  n++;
182  }
183  if (startPoint->getVertexType() == svCorner) {
184  if (startPoint->numberOfFaces() > 1 && n == 2)
185  startPoint->setVertexType(svCrease);
186  }
187  else {
188  if (n == 0)
189  startPoint->setVertexType(svRegular);
190  else if (n == 1)
191  startPoint->setVertexType(svDart);
192  else if (n == 2)
193  startPoint->setVertexType(svCrease);
194  else if (n > 2)
195  startPoint->setVertexType(svCorner);
196  }
197  n = 0;
198  for (size_t i=0; i<endPoint->numberOfEdges(); ++i) {
199  SubdivisionEdge* edge = endPoint->getEdge(i);
200  if (edge->isCrease())
201  n++;
202  }
203  if (endPoint->getVertexType() == svCorner) {
204  if (endPoint->numberOfFaces() > 1 && n == 2)
205  endPoint->setVertexType(svCrease);
206  }
207  else {
208  if (n == 0)
209  endPoint->setVertexType(svRegular);
210  else if (n == 1)
211  endPoint->setVertexType(svDart);
212  else if (n == 2)
213  endPoint->setVertexType(svCrease);

342  else {
343  // p2 lies on port
344  vertices << p1;
345  vertices << m;
346  // p2 lies on stbd
347  p2.setY(-p2.y());
348  m.setY(-m.y());
349  vertices << p2;

525 SubdivisionControlPoint* SubdivisionControlEdge::insertControlPoint(const QVector3D& p)
526 {
527  SubdivisionControlPoint* result = SubdivisionControlPoint::construct(_owner);
528  SubdivisionControlPoint* sp = dynamic_cast<SubdivisionControlPoint*>(startPoint());
529  SubdivisionControlPoint* ep = dynamic_cast<SubdivisionControlPoint*>(endPoint());
530  if (sp == nullptr || ep == nullptr)
531  throw invalid_argument("start and/or end point are not control points");
532  result->setCoordinate(p);
533  if (getCurve() != 0) {
534  // insert the new point in the controlcurve
535  getCurve()->insertControlPoint(sp, ep, result);
536  }
537  _owner->addControlPoint(result);
538  for (size_t i=0; i<_faces.size(); ++i) {
539  SubdivisionFace* face = _faces[i];
540  if (face->hasPoint(sp) && face->hasPoint(ep)) {
541  size_t i1 = face->indexOfPoint(sp);
542  size_t i2 = face->indexOfPoint(ep);
543  if (i2 == i1+1)
544  face->insertPoint(i2, result);
545  else if (i1 == i2+1)
546  face->insertPoint(i1, result);
547  else if (i1 == 0 && i2 == face->numberOfPoints() - 1)
548  face->insertPoint(0, result);
549  else if (i2 == 0 && i1 == face->numberOfPoints() - 1)
550  face->insertPoint(0, result);
551  result->addFace(face);
552  }
553  }
554  ep->deleteEdge(this);
555  SubdivisionControlEdge* edge = _owner->addControlEdge(result, ep);
556  edge->setCrease(isCrease());
557  edge->setCurve(getCurve());
558  if (getCurve())
559  result->setVertexType(svCrease);
560  for (size_t i=0; i<_faces.size(); ++i)
561  edge->addFace(_faces[i]);
562  _points[1] = result;
563  result->addEdge(this);

674  else {
675  glLineWidth(1);
676  }
677  if (vp.getViewportType() == fvBodyplan) {
678  float mfl = _owner->getMainframeLocation();
679  if ((p1.x() < mfl && p2.x() > mfl) || (p1.x() > mfl && p2.x() < mfl)) {
680  // straddles mainframe
681  QVector3D m;
682  if (p2.x() - p1.x() != 0)
683  m = Interpolate(p1, p2, (mfl - p1.x()) / (p2.x() - p1.x()));
684  else
685  m = MidPoint(p1, p2);
686  if (p1.x() <= mfl) {
687  // p2 lies on port
688  vertices << p2;
689  vertices << m;
690  // p1 lies on stbd
691  p1.setY(-p1.y());
692  m.setY(-m.y());
693  vertices << p1;
694  vertices << m;
695  }
696  else {
697  // p2 lies on port
698  vertices << p1;
699  vertices << m;
700  // p2 lies on stbd
701  p2.setY(-p2.y());
702  m.setY(-m.y());
703  vertices << p2;
704  vertices << m;
705  }
706  }
707  else {
708  if (p1.x() <= mfl)
709  p1.setY(-p1.y());
710  if (p2.x() <= mfl)
711  p2.setY(-p2.y());
712  vertices << p1;
713  vertices << p2;
714  }
715  }
716  else {
717  vertices << p1;
718  vertices << p2;
719  }
720  }
721  else {
722  // viewport is not vmWireFrame, drawing to Z buffer
723  glLineWidth(1);
724  vertices << p1;
725  vertices << p2;
726  }
727  lineshader->renderLines(vertices, getColor());
728 }
