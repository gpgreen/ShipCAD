79 static float triangle_area(const QVector3D& p1, const QVector3D& p2, const QVector3D& p3)
80 {
81  float ax = 0.5 * ((p1.y() - p2.y()) * (p1.z() + p2.z())
82  + (p2.y() - p3.y()) * (p2.z() + p3.z())
83  + (p3.y() - p1.y()) * (p3.z() + p1.z()));
84  float ay = 0.5 * ((p1.z() - p2.z()) * (p1.x() + p2.x())
85  + (p2.z() - p3.z()) * (p2.x() + p3.x())
86  + (p3.z() - p1.z()) * (p3.x() + p1.x()));
87  float az = 0.5 * ((p1.x() - p2.x()) * (p1.y() + p2.y())
88  + (p2.x() - p3.x()) * (p2.y() + p3.y())
89  + (p3.x() - p1.x()) * (p3.y() + p1.y()));
90  return sqrt(ax*ax + ay*ay + az*az);
91 }

167 // algorithm from http://geomalgorithms.com/a06-_intersect.html
168 static bool privIntersectWithRay(const PickRay& ray, const QVector3D& p1, const QVector3D& p2,
169  const QVector3D& p3)
170 {
171  // get triangle edge vectors and plane normal
172  QVector3D u = p2 - p1;
173  QVector3D v = p3 - p1;
174  QVector3D n = QVector3D::crossProduct(u, v);
175  // check for degenerate triangle
176  if (n.x() == 0 && n.y() == 0 && n.z() == 0)
177  return false;
178  QVector3D w0 = ray.pt - p1;
179  float a = -QVector3D::dotProduct(n, w0);
180  float b = QVector3D::dotProduct(n, ray.dir);
181  if (abs(b) < 1E-5) {
182  if (a == 0)
183  return true; // ray lies in triangle plane
184  return false; // ray disjoint from plane
185  }
186 
187  // get intersect point of ray with triangle plane
188  float r = a / b;
189  if (r < 0.0) // ray goes away from triangle
190  return false;
191 
192  QVector3D pt = ray.pt + r * ray.dir; // intersect point of ray and plane
193 
194  // is point in triangle?
195  return PointInTriangle(pt, p1, p2, p3);
196 }

515 // function used during zebra shading, splits triangle up based on coloring
516 static void zebraProcess(const QVector3D& p, float dp,
517  vector<ZebraIntersection>& intersections)
518 {
519  float prevdp = intersections.back().dotprod;
520  float lf;
521  modf(prevdp/Width, &lf);
522  float hf;
523  modf(dp/Width, &hf);
524  int lo = static_cast<int>(lf);
525  int hi = static_cast<int>(hf);
526  if ((hi - lo) > 0) {
527  const QVector3D& prevp = intersections.back().p;
528  for (int i=lo; i<=hi; i++) {
529  float val = i * Width;
530  if (val > prevdp && val < dp) {
531  float t = (val - prevdp) / (dp - prevdp);
532  QVector3D p3d = prevp + t * (p - prevp);
533  intersections.push_back(ZebraIntersection(p3d, val));
534  }
535  }
536  intersections.push_back(ZebraIntersection(p, dp));
537  } else if ((lo - hi) > 0) {
538  const QVector3D& prevp = intersections.back().p;
539  for (int i=lo; i>=hi; i--) {
540  float val = i * Width;
541  if (val < prevdp && val > dp) {
542  float t = (val - prevdp) / (dp - prevdp);
543  QVector3D p3d = prevp + t * (p - prevp);
544  intersections.push_back(ZebraIntersection(p3d, val));
545  }
546  }
547  intersections.push_back(ZebraIntersection(p, dp));
548  } else
549  intersections.push_back(ZebraIntersection(p, dp));
550 }
