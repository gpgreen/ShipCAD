97 void SubdivisionLayer::processTriangle(const QVector3D& p1,
98  const QVector3D& p2,
99  const QVector3D& p3,
100  LayerProperties& props) const
101 {
102  QVector3D center = (p1 + p2 + p3) / 3.0f;
103  float ax = 0.5 * ((p1.y() - p2.y()) * (p1.z() + p2.z())
104  + (p2.y() - p3.y()) * (p2.z() + p3.z())
105  + (p3.y() - p1.y()) * (p3.z() + p1.z()));
106  float ay = 0.5 * ((p1.z() - p2.z()) * (p1.x() + p2.x())
107  + (p2.z() - p3.z()) * (p2.x() + p3.x())
108  + (p3.z() - p1.z()) * (p3.x() + p1.x()));
109  float az = 0.5 * ((p1.x() - p2.x()) * (p1.y() + p2.y())
110  + (p2.x() - p3.x()) * (p2.y() + p3.y())
111  + (p3.x() - p1.x()) * (p3.y() + p1.y()));
112  float area = sqrt(ax * ax + ay * ay + az * az);
113  props.surface_area += area;
114  props.surface_center_of_gravity += (area * center);
115 }

117 LayerProperties SubdivisionLayer::getSurfaceProperties() const
118 {
119  LayerProperties result;
120  result.surface_area = 0;
121  result.weight = 0;
122  result.surface_center_of_gravity = ZERO;
123 
124  for (size_t i=1; i<=_patches.size(); ++i) {
125  for (size_t j=1; j<=_patches[i-1]->numberOfChildren(); ++j) {
126  SubdivisionFace* child = _patches[i-1]->getChild(j-1);
127  for (size_t k=3; k<=child->numberOfPoints(); ++k) {
128  processTriangle(child->getPoint(0)->getCoordinate(),
129  child->getPoint(k-2)->getCoordinate(),
130  child->getPoint(k-1)->getCoordinate(),
131  result);
132  }
133  }
134  }
135  if (result.surface_area != 0) {
136  result.surface_center_of_gravity /= result.surface_area;
137  if (isSymmetric()) {
138  result.surface_area *= 2.0f;
139  result.surface_center_of_gravity.setY(0);
140  }
141  result.weight = result.surface_area * _thickness * _material_density;
142  }
143  return result;
144 }

322 bool SubdivisionLayer::calculateIntersectionPoints(SubdivisionLayer* layer)
323 {
324  bool result = false;
325  vector<SubdivisionControlEdge*> edges;
326  vector<SubdivisionControlPoint*> newpoints;
327 
328  SubdivisionControlPoint* p1, *p2;
329  SubdivisionControlEdge* edge;
330  SubdivisionControlFace* face;
331 
332  // assemble all controledges in a list
333  for (size_t i=1; i<=_patches.size(); ++i) {
334  face = _patches[i-1];
335  p1 = dynamic_cast<SubdivisionControlPoint*>(face->getPoint(face->numberOfPoints()-1));
336  for (size_t j=1; j<=face->numberOfPoints(); ++j) {
337  p2 = dynamic_cast<SubdivisionControlPoint*>(face->getPoint(j-1));
338  edge = _owner->controlEdgeExists(p1, p2);
339  if (edge != 0 && find(edges.begin(), edges.end(), edge) == edges.end())
340  edges.push_back(edge);
341  p1 = p2;
342  }
343  }
344 
345  // now check all edges for intersection with layer 2
346  size_t i = 1;
347  while (i <= edges.size()) {
348  edge = edges[i-1];
349  bool intfound = false;
350  size_t j = 1;
351  while (j <= layer->numberOfFaces() && !intfound) {
352  face = layer->getFace(j-1);
353  size_t k = 1;
354  while (k <= face->numberOfChildren() && !intfound) {
355  SubdivisionFace* child = face->getChild(k-1);
356  size_t l = 3;
357  while (l <= child->numberOfPoints() && !intfound) {
358  Plane pln(child->getPoint(0)->getCoordinate(),
359  child->getPoint(l-2)->getCoordinate(),
360  child->getPoint(l-1)->getCoordinate());
361  float s1 = pln.a() * edge->startPoint()->getCoordinate().x()
362  + pln.b() * edge->startPoint()->getCoordinate().y()
363  + pln.c() * edge->startPoint()->getCoordinate().z()
364  + pln.d();
365  float s2 = pln.a() * edge->endPoint()->getCoordinate().x()
366  + pln.b() * edge->endPoint()->getCoordinate().y()
367  + pln.c() * edge->endPoint()->getCoordinate().z()
368  + pln.d();
369  if ((s1 < 0 && s2 > 0) || (s1 > 0 && s2 < 0)) {
370  // edge intersects the plane, does it lie in the triangle?
371  float t;
372  if (s1 == s2)
373  t = 0.5;
374  else
375  t = -s1 / (s2 - s1);
376  QVector3D p3d = edge->startPoint()->getCoordinate()
377  + t * (edge->endPoint()->getCoordinate()
378  - edge->startPoint()->getCoordinate());
379  if (PointInTriangle(p3d, child->getPoint(0)->getCoordinate(),
380  child->getPoint(l-2)->getCoordinate(),
381  child->getPoint(l-1)->getCoordinate())) {
382  // yes, valid intersection
383  SubdivisionControlPoint* p = edge->insertControlPoint(p3d);
384  if (p != 0) {
385  intfound = true;
386  result = true;
387  newpoints.push_back(p);
388  }
389  }
390  }
391  l++;
392  }
393  k++;
394  }
395  j++;
396  }
397  i++;
398  }
399  if (newpoints.size() > 0) {
400  // try to find multiple new points belonging to the same face and insert an edge
401 restart:
402  size_t i = 0;
403  edges.clear();
404  bool inserted = false;
405  while (i < newpoints.size()) {
406  p1 = newpoints[i];
407  size_t j = 0;
408  while (j < p1->numberOfFaces()) {
409  face = dynamic_cast<SubdivisionControlFace*>(p1->getFace(j));
410  size_t k = 0;
411  while (k < face->numberOfPoints() && !inserted) {
412  p2 = dynamic_cast<SubdivisionControlPoint*>(face->getPoint(k));
413  if (p1 != p2 && find(newpoints.begin(), newpoints.end(), p2) != newpoints.end()) {
414  // this is also a new point, first check if an edge already exists between p1 and p2
415  if (_owner->edgeExists(p1, p2) == 0) {
416  inserted = true;
417  bool deleteface;
418  edge = face->insertControlEdge(p1, p2, deleteface);
419  edge->setSelected(true);
420  edges.push_back(edge);
421  // if face has been deleted, restart the loop
422  if (deleteface) {
423  getOwner()->deleteControlFace(face);
424  goto restart;
425  }
426  }
427  }
428  k++;
429  }
430  if (!inserted)
431  j++;
432  }
433  i++;
434  }
435  }
436  return result;
437 }

536 void SubdivisionLayer::unroll(PointerVector<DevelopedPatch>& destination)
537 {
538  vector<SubdivisionControlFace*> todo(_patches.begin(), _patches.end());
539  vector<vector<SubdivisionControlFace*> > done;
540 
541  while (todo.size() > 0) {
542  SubdivisionControlFace* face = todo.back();
543  todo.pop_back();
544  vector<SubdivisionControlFace*> current;
545  current.push_back(face);
546  findAttachedFaces(current, face, todo);

551  vector<SubdivisionControlFace*>& current = done[i];

555  QString str;
556  if (done.size() == 1)
557  str = getName();
558  else
559  str = QString("%1 %2 %3").arg(getName()).arg("Part").arg(i+1);
560  patch->setName(str);
561  destination.add(patch);
562  if (!patch->isMirror() && isSymmetric()) {
563  QString nm(str);
564  nm += " (SB)";
565  patch->setName(nm);
566  DevelopedPatch* copy = new DevelopedPatch(this);
567  QString nm1(str);

712 void SubdivisionLayer::saveToDXF(QStringList& strings)
713 {
714  if (!isVisible())
715  return;
716  vector<ControlFaceGrid> assembled;
717  getOwner()->assembleFacesToPatches(amRegular, assembled);
718  if (assembled.size() == 0)
719  return;
720  for (size_t i=0; i<assembled.size(); ++i) {
721  ControlFaceGrid assface = assembled[i];
722  if ((assface.cols() > 1 && assface.rows() >= 1) ||
723  (assface.cols() >= 1 && assface.rows() > 1) ||
724  (assface.cols() == 1 && assface.rows() == 1 && assface.getFace(0,0)->numberOfPoints() == 4)) {
725  PointGrid grid;
726  getOwner()->convertToGrid(assface, grid);
727  if (grid.cols() > 0 && grid.rows() > 0) {
728  strings.push_back(QString("0\r\nPOLYLINE"));
729  strings.push_back(QString("8\r\n%1").arg(getName()));
730  strings.push_back(QString("62\r\n%1").arg(FindDXFColorIndex(getColor())));
731  strings.push_back(QString("66\r\n1"));
732  strings.push_back(QString("70\r\n16"));
733  strings.push_back(QString("71\r\n%1").arg(grid.rows()));
734  strings.push_back(QString("72\r\n%1").arg(grid.cols()));
735  for (size_t j=0; j<grid.rows(); j++) {
736  for (size_t k=0; k<grid.cols(); k++) {
737  strings.push_back(QString("0\r\nVERTEX"));
738  strings.push_back(QString("8\r\n%1").arg(getName()));
739  QVector3D p = grid.getPoint(j, k)->getCoordinate();
740  strings.push_back(QString("10\r\n%1").arg(Truncate(p.x(), 4)));
741  strings.push_back(QString("20\r\n%1").arg(Truncate(p.y(), 4)));
742  strings.push_back(QString("30\r\n%1").arg(Truncate(p.z(), 4)));
743  strings.push_back(QString("70\r\n64")); // polygon mesh vertex
744  }
745  }
746  strings.push_back(QString("0\r\nSEQEND"));
747  if (isSymmetric() && getOwner()->drawMirror()) {
748  strings.push_back(QString("0\r\nPOLYLINE"));
749  strings.push_back(QString("8\r\n%1").arg(getName()));
750  strings.push_back(QString("62\r\n%1").arg(FindDXFColorIndex(getColor())));
751  strings.push_back(QString("66\r\n1"));

760  strings.push_back(QString("10\r\n%1").arg(Truncate(p.x(), 4)));
761  strings.push_back(QString("20\r\n%1").arg(Truncate(-p.y(), 4)));
762  strings.push_back(QString("30\r\n%1").arg(Truncate(p.z(), 4)));
763  strings.push_back(QString("70\r\n64")); // polygon mesh vertex
764  }
765  }
766  strings.push_back(QString("0\r\nSEQEND"));
767  }
