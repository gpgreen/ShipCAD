107 float SubdivisionPoint::Angle_VV_3D(const QVector3D& p1,
108  const QVector3D& p2,
109  const QVector3D& p3) const
110 {
111  QVector3D v1 = p1 - p2;
112  QVector3D v2 = p3 - p2;
113  v1.normalize();
114  v2.normalize();
115  float l = v1.x()*v2.x() + v1.y()*v2.y() + v1.z()*v2.z();
116  if (l < -1)
117  l = -1;
118  else if (l > 1)
119  l = 1;
120  return acos(l);
121 }

123 float SubdivisionPoint::getCurvature() const
124 {
125  float result = 0.0;
126  for (size_t i=0; i<_edges.size(); ++i)
127  if (_edges[i]->numberOfFaces() < 2)
128  return result;
129  float sigma = 0;
130  for (size_t j=0; j<_faces.size(); ++j) {
131  SubdivisionFace* face = _faces[j];
132  size_t index = face->indexOfPoint(this);
133  size_t previndex = (index + face->numberOfPoints() - 1) % face->numberOfPoints();
134  size_t nextindex = (previndex + 2) % face->numberOfPoints();
135  SubdivisionPoint* prev = face->getPoint(previndex);
136  SubdivisionPoint* next = face->getPoint(nextindex);
137  float tmp = Angle_VV_3D(prev->getCoordinate(),
138  _coordinate,
139  next->getCoordinate());
140  sigma += tmp;
141  }
142  result = (360 - RadToDeg(sigma));
143  return result;
144 }

207 QVector3D SubdivisionPoint::getLimitPoint() const
208 {
209  QVector3D result = ZERO;
210  SubdivisionPoint* p30 = 0;
211  SubdivisionPoint* p33 = 0;
212  SubdivisionPoint* p = 0;
213  if (_vtype == svDart || _vtype == svRegular) {
214  result = ZERO;
215  size_t n = _faces.size();
216  for (size_t i=0; i<n; ++i) {
217  SubdivisionFace* face = _faces[i];
218  size_t ind = face->indexOfPoint(this);
219  p30 = face->getPoint((ind + 1) % face->numberOfPoints());
220  p33 = face->getPoint((ind + 2) % face->numberOfPoints());
221  result += (n * _coordinate + 4 * p30->getCoordinate() + p33->getCoordinate());
222  }
223  result /= (n * (n + 5));
224  }
225  else if (_vtype == svCrease) {
226  for (size_t i=0; i<_edges.size(); ++i) {
227  SubdivisionEdge* edge = _edges[i];
228  if (edge->isCrease()) {
229  if (edge->startPoint() == this)
230  p = edge->endPoint();
231  else
232  p = edge->startPoint();
233  if (p30 == 0)
234  p30 = p;
235  else
236  p33 = p;
237  }
238  }
239  if (p30 != 0 && p33 != 0) {
240  result = (1/6.0f)*p30->getCoordinate()
241  + (2/3.0f)*result
242  + (1/6.0f)*p33->getCoordinate();
243  }
244  else {
245  // this is an error
246  result = getCoordinate();
247  }
248  }
249  else
250  result = getCoordinate();
251  return result;
252 }

333 QVector3D SubdivisionPoint::averaging() const
334 {
335  QVector3D result;
336  SubdivisionPoint* p;
337  float totalweight = 0.0;
338  size_t nt = 0;
339  float weight;
340 
341  if (_edges.size() == 0 || _vtype == svCorner)
342  result = getCoordinate();
343  else {
344  if (_vtype == svCrease) {
345  result = getCoordinate() * 0.5;
346  for (size_t i=0; i<_edges.size(); ++i) {
347  SubdivisionEdge* edge = _edges[i];
348  if (edge->numberOfFaces() == 1 || edge->isCrease()) {
349  if (edge->startPoint() == this)
350  p = edge->endPoint();
351  else
352  p = edge->startPoint();
353  result += (0.25 * p->getCoordinate());
354  }
355  }
356  }
357  else {
358  result = ZERO;
359  for (size_t i=0; i<_faces.size(); ++i) {
360  SubdivisionFace* face = _faces[i];
361  QVector3D center = ZERO;
362  if (face->numberOfPoints() == 3) {
363  ++nt;
364  // calculate centerpoint
365  for (size_t j=0; j<face->numberOfPoints(); ++j) {
366  p = face->getPoint(j);
367  if (p == this)
368  weight = .25;
369  else
370  weight = .375;
371  center += (weight * p->getCoordinate());
372  }
373  weight = third_pi;
374  }
375  else if (face->numberOfPoints() == 4) {
376  for (size_t j=0; j<face->numberOfPoints(); ++j) {
377  p = face->getPoint(j);
378  weight = .25;
379  center += (weight * p->getCoordinate());
380  }
381  weight = half_pi;
382  }
383  else
384  throw runtime_error("invalid number of points in SubdivisionPoint::averaging");
385  result += (weight * center);
386  totalweight += weight;
387  }
388  if (totalweight != 0) {
389  result /= totalweight;
390  }
391  size_t nq = _faces.size() - nt;
392  float a;
393  if (nt == _faces.size()) {
394  // apply averaging in case of vertex surrounded by triangles
395  a = 5/3.0 - 8/3.0*sqrt(.375+.25*cos(two_pi/_faces.size()));
396  }
397  else if (nq == _faces.size()) {
398  // apply averaging in case of vertex surrounded by quads
399  a = 4 / static_cast<float>(_faces.size());
400  }
401  else {
402  // apply averaging in case of vertex on boundary of quads and triangles
403  if (nq == 0 && nt == 3)
404  a = 1.5;
405  else
406  a = 12 / static_cast<float>(3 * nq + 2 * nt);
407  }
408  if (a != 1.0) {
409  result = getCoordinate() + a * (result - getCoordinate());
410  }
411  }
412  }
413  return result;
414 }

463 QVector3D SubdivisionPoint::getNormal() const
464 {
465  QVector3D result;
466  for (size_t i=0; i<_faces.size(); ++i) {
467  SubdivisionFace* face = _faces[i];
468  if (face->numberOfPoints() > 4) {
469  // face possibly concave at this point
470  // use the normal of all points from this face
471  QVector3D c = face->getFaceCenter();
472  QVector3D n = ZERO;
473  for (size_t j=1; j<face->numberOfPoints(); ++j) {
474  n = n + UnifiedNormal(c, face->getPoint(j-1)->getCoordinate(),
475  face->getPoint(j)->getCoordinate());
476  }
477  n.normalize();
478  result = result + n;
479  }
480  else {
481  size_t index = face->indexOfPoint(this);
482  size_t j = (index + face->numberOfPoints() - 1) % face->numberOfPoints();
483  QVector3D p1 = face->getPoint(j)->getCoordinate();
484  j = (index + face->numberOfPoints() + 1) % face->numberOfPoints();
485  QVector3D p3 = face->getPoint(j)->getCoordinate();
486  result = result + UnifiedNormal(p1, getCoordinate(), p3);
487  }
488  }
489  result.normalize();
490  return result;
491 }

584  // neighbouring controlfaces belongs to a visible layer

586  if (_owner->showControlNet()) {

589  if (face == 0)
590  continue;

592  if (face->getLayer()->isVisible())
593  return true;
594  }
595  }
596  }
597  // finally check if the point is selected
598  // selected points must be visible at all times
599  // points with no faces connected also!
600  result = isSelected() || _faces.size() == 0;
601  if (!result && _owner->numberOfControlCurves() > 0) {
602  for (size_t i=0; i<numberOfEdges(); ++i) {
603  if (_edges[i]->getCurve() != 0) {
604  if (_edges[i]->getCurve()->isSelected()) {
605  result = true;
606  break;
607  }
608  }
609  }
610  }
611  return result;

643 {
644  SubdivisionControlPoint* p1, *p2;
645  SubdivisionControlEdge* edge1, *edge2;
646  SubdivisionControlFace* face;
647  if (_faces.size() <= 2) {
648  setSelected(false);
649  p1 = nullptr;
650  p2 = nullptr;
651  // this is possibly a point on a boundary edge
652  // check for this special case
653  edge1 = nullptr;
654  edge2 = nullptr;
655  for (size_t i=0; i<numberOfEdges(); ++i) {
656  if (_edges[i]->numberOfFaces() == 1) {
657  if (edge1 == 0) {
658  edge1 = dynamic_cast<SubdivisionControlEdge*>(_edges[i]);
659  }
660  else {
661  edge2 = dynamic_cast<SubdivisionControlEdge*>(_edges[i]);
662  }
663  }
664  }
665  if (edge1 != nullptr && edge2 != nullptr) {
666  _owner->collapseEdge(edge2);
667  _owner->collapseEdge(edge1);
668  }
669  bool edge_collapse = false;
670  bool crease = false;
671  if (_edges.size() == 2) {
672  edge_collapse = true;
673  edge1 = dynamic_cast<SubdivisionControlEdge*>(_edges[0]);
674  if (edge1 == nullptr)
675  throw invalid_argument("edge is not a control edge SubdivisionControlPoint::collapse");
676  if (edge1->startPoint() == this) {
677  p1 = dynamic_cast<SubdivisionControlPoint*>(edge1->endPoint());
678  }
679  else {
680  p1 = dynamic_cast<SubdivisionControlPoint*>(edge1->startPoint());
681  }
682  edge2 = dynamic_cast<SubdivisionControlEdge*>(_edges[1]);
683  if (edge2 == nullptr)
684  throw invalid_argument("edge is not a control edge SubdivisionControlPoint::collapse");
685  if (edge2->startPoint() == this) {
686  p2 = dynamic_cast<SubdivisionControlPoint*>(edge2->endPoint());
687  }
688  else {
689  p2 = dynamic_cast<SubdivisionControlPoint*>(edge2->startPoint());
690  }
691  crease = edge1->isCrease() || edge2->isCrease();
692  }
693  for (size_t i=numberOfFaces(); i>0; --i) {
694  face = dynamic_cast<SubdivisionControlFace*>(_faces[i-1]);
695  vector<SubdivisionControlPoint*> points;

697  SubdivisionControlPoint* pt = dynamic_cast<SubdivisionControlPoint*>(face->getPoint(j));

700  }
701  _owner->addControlFace(points, false, face->getLayer());
702  _owner->deleteControlFace(face);
703  }
704 
705  if (edge_collapse) {
706  edge1 = _owner->controlEdgeExists(p1, p2);
707  if (edge1 != nullptr)
708  edge1->setCrease(crease);
709  }
710  } // faces.size <= 2
711  else {
712  vector<SubdivisionControlPoint*> checklist;
713  vector<SubdivisionControlEdge*> edges;
714  for (size_t i=0; i<numberOfEdges(); ++i) {
715  SubdivisionControlPoint* s = dynamic_cast<SubdivisionControlPoint*>(edges[i]->startPoint());
716  SubdivisionControlPoint* e = dynamic_cast<SubdivisionControlPoint*>(edges[i]->endPoint());
717  if (s == nullptr || e == nullptr)
718  throw invalid_argument("point not control point SubdivisionControlPoint::collapse");
719  if (s == this)
720  checklist.push_back(e);
721  else
722  checklist.push_back(s);
723  }
724  for (size_t i=0; i<numberOfFaces(); ++i) {
725  face = dynamic_cast<SubdivisionControlFace*>(_faces[i]);
726  if (face == nullptr)
727  throw invalid_argument("face is not a control face SubdivisionControlPoint::collapse");
728  p1 = dynamic_cast<SubdivisionControlPoint*>(face->getPoint(face->numberOfPoints()-1));
729  if (p1 == nullptr)
730  throw invalid_argument("point not control point SubdivisionControlPoint::collapse");
731  for (size_t j=0; j<face->numberOfPoints(); ++j) {
732  p2 = dynamic_cast<SubdivisionControlPoint*>(face->getPoint(j));
733  if (p2 == nullptr)
734  throw invalid_argument("point not control point SubdivisionControlPoint::collapse");
735  if (p1 != this && p2 != this) {
736  edge1 = _owner->controlEdgeExists(p1, p2);
737  if (edge1 != nullptr
738  && find(edges.begin(), edges.end(), edge1) == edges.end())
739  _edges.push_back(edge1);
740  }
741  p1 = p2;
742  }
743  }
744  // sort edges in correct order and add new face
745  if (edges.size() > 2) {
746  vector<vector<SubdivisionControlPoint*> > sorted;
747  _owner->isolateEdges(edges, sorted);
748  for (size_t i=0; i<sorted.size(); ++i) {
749  vector<SubdivisionControlPoint*>& points = sorted[i];
750  if (points.size() > 2) {
751  face = _owner->addControlFace(points, false);
752  if (face != nullptr) {
753  p1 = dynamic_cast<SubdivisionControlPoint*>(face->getPoint(face->numberOfPoints()-1));
754  if (p1 == nullptr)
755  throw invalid_argument("point not control point SubdivisionControlPoint::collapse");
756  for (size_t j=0; j<face->numberOfPoints(); j++) {
757  p2 = dynamic_cast<SubdivisionControlPoint*>(face->getPoint(j));
758  if (p2 == nullptr)
759  throw invalid_argument("point not control point SubdivisionControlPoint::collapse");
760  // BUGBUG: doesn't do anything here
761  edge1 = _owner->controlEdgeExists(p1, p2);
762  p1 = p2;
763  }
764  }
765  }
766  }
767  }
768  // BUGBUG: supposed to delete point here
769  for (size_t i=checklist.size(); i>0; --i) {
770  p1 = checklist[i-1];
771  if (p1->numberOfFaces() > 1 && p1->numberOfEdges() == 2)
772  p1->collapse();
773  }
774  }
