173 SubdivisionControlPoint* SubdivisionSurface::addControlPoint(const QVector3D& pt)
174 {
175  SubdivisionControlPoint* result = 0;
176  double max_error = 1E-5;
177  for (size_t i=0; i<numberOfControlEdges(); ++i) {
178  SubdivisionControlEdge* edge = getControlEdge(i);
179  if (edge->numberOfFaces() <= 1) { // boundary edge
180  if (SquaredDistPP(pt, edge->startPoint()->getCoordinate()) <= max_error) {
181  result = dynamic_cast<SubdivisionControlPoint*>(edge->startPoint());
182  break;
183  }
184  else if (SquaredDistPP(pt, edge->endPoint()->getCoordinate()) <= max_error) {
185  result = dynamic_cast<SubdivisionControlPoint*>(edge->endPoint());
186  break;
187  }
188  }
189  }
190  if (result == 0) {
191  // search controlpoints without edges
192  for (size_t i=0; i<numberOfControlPoints(); ++i) {
193  SubdivisionControlPoint* point = getControlPoint(i);
194  if (point->numberOfEdges() == 0) {
195  if (SquaredDistPP(pt, point->getCoordinate()) <= max_error) {
196  result = point;
197  break;
198  }
199  }
200  }
201  }
202  if (result == 0) {
203  result = newControlPoint(pt);
204  }
205  return result;
206 }
