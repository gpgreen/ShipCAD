133 int ShipCAD::FindDXFColorIndex(QColor color)
134 {
135  int result = 1;
136  int r = color.red();
137  int g = color.green();
138  int b = color.blue();
139  float dist = 1E30f;
140  for (int i=0; i<255; ++i) {
141  int r1 = DXFLayerColors[i].red();
142  int g1 = DXFLayerColors[i].green();
143  int b1 = DXFLayerColors[i].blue();
144  float tmp = sqrt(static_cast<float>((r1-r)*(r1-r) + (g1-g)*(g1-g) + (b1-b)*(b1-b)));
145  if (tmp < dist) {
146  dist = tmp;
147  result = i+1;
148  }
149  }
150  return result;
151 }

160 QColor ShipCAD::FillColor(float parameter)
161 {

163

168  int trunc_wavelength = static_cast<int>(ftrunc_wavelength);

171  green = 0.0;
172  blue = 1.0;
173  }
174  else if (trunc_wavelength >= 440 && trunc_wavelength <= 489) {
175  red = 0.0;
176  green = (wavelength - 440.0) / (490.0 - 440.0);

178  }
179  else if (trunc_wavelength >= 490 && trunc_wavelength <= 509) {
180  red = 0.0;
181  green = 1.0;

253 static bool SameSide(const QVector3D& p1,
254  const QVector3D& p2,
255  const QVector3D& a,
256  const QVector3D& b)
257 {
258  QVector3D ba = b - a;
259  QVector3D p = p1 - a;
260  QVector3D cp1 = QVector3D::crossProduct(p, ba);
261  p = p2 - a;
262  QVector3D cp2 = QVector3D::crossProduct(p, ba);
263  float dp = QVector3D::dotProduct(cp1, cp2);
264  if (fabs(dp) < 1E5f)
265  dp = dp - 1 + 1;
266  return dp >= 0;
267 }

278 // clip a triangle given the 3 distances from a plane,
279 // returns 2 sets of coordinates, front of the plane, and back of the plane
280 void ShipCAD::ClipTriangle(const QVector3D& p1,
281  const QVector3D& p2,
282  const QVector3D& p3,
283  float s1,
284  float s2,
285  float s3,
286  std::vector<QVector3D>& front,
287  std::vector<QVector3D>& back)
288 {
289  if (s1 <=0 && s2 <= 0 && s3 <= 0) {
290  // all at the back of the plane
291  back.push_back(p1);
292  back.push_back(p2);
293  back.push_back(p3);
294  }
295  else if (s1 > 0 && s2 > 0 && s3 > 0) {
296  // all at the front of the plane
297  front.push_back(p1);
298  front.push_back(p2);
299  front.push_back(p3);
300  }
301  else {
302  // triangle spans the plane, calculate the intersections
303  if (s1 <= 0)
304  back.push_back(p1);
305  if (s1 >= 0)
306  front.push_back(p1);
307  if ((s1 < 0 && s2 > 0) || (s1 > 0 && s2 < 0)) {
308  float t;
309  if (s1 == s2)
310  t = 0.5;
311  else
312  t = -s1 / (s2 - s1);
313  QVector3D p = p1 + (t * (p2 - p1));
314  back.push_back(p);
315  front.push_back(p);
316  }
317  if (s2 <= 0)
318  back.push_back(p2);
319  if (s2 >= 0)
320  front.push_back(p2);
321  if ((s2 < 0 && s3 > 0) || (s2 > 0 && s3 < 0)) {
322  float t;
323  if (s2 == s3)
324  t = 0.5;
325  else
326  t = -s2 / (s3 - s2);
327  QVector3D p = p2 + (t * (p3 - p2));
328  back.push_back(p);
329  front.push_back(p);
330  }
331  if (s3 <= 0)
332  back.push_back(p3);
333  if (s3 >= 0)
334  front.push_back(p3);
335  if ((s3 < 0 && s1 > 0) || (s3 > 0 && s1 < 0)) {
336  float t;
337  if (s2 == s3)
338  t = 0.5;
339  else
340  t = -s3 / (s1 - s3);
341  QVector3D p = p3 + (t * (p1 - p3));
342  back.push_back(p);
343  front.push_back(p);
344  }
345  }
346 }

375 bool ShipCAD::Lines3DIntersect(const QVector3D& p1, const QVector3D& p2, const QVector3D& p3,
376  const QVector3D& p4, double& param, QVector3D& int1)
377 {
378  bool result = false;
379  param = 0;
380  int1 = ZERO;
381  QVector3D p13 = p1 - p3;
382  QVector3D p43 = p4 - p3;
383  if (abs(p43.x()) < 1E-6 && abs(p43.y()) < 1E-6 && abs(p43.z()) < 1E-6)
384  return result;
385  QVector3D p21 = p2 - p1;
386  if (abs(p21.x()) < 1E-6 && abs(p21.y()) < 1E-6 && abs(p21.z()) < 1E-6)
387  return result;
388  double d1343 = p13.x() * p43.x() + p13.y() * p43.y() + p13.z() * p43.z();
389  double d4321 = p43.x() * p21.x() + p43.y() * p21.y() + p43.z() * p21.z();
390  double d1321 = p13.x() * p21.x() + p13.y() * p21.y() + p13.z() * p21.z();
391  double d4343 = p43.x() * p43.x() + p43.y() * p43.y() + p43.z() * p43.z();
392  double d2121 = p21.x() * p21.x() + p21.y() * p21.y() + p21.z() * p21.z();
393 
394  double denom = d2121 * d4343 - d4321 * d4321;
395  if (abs(denom) < 1E-6)
396  return result;
397  double numer = d1343 * d4321 - d1321 * d4343;
398  double mua = numer / denom;
399  double mub = (d1343 + d4321 * mua) / d4343;
400  QVector3D pa = p1 + mua * p21;
401  QVector3D pb = p3 + mub * p43;
402  if (mua > -1E-6 && mua < 1 && mub > -1E-6 && mub < 1) {
403  result = true;
404  int1 = pa;
405  param = mua;
406  }
407  return result;
408 }

427 // This procedure takes a lot of linesegments and tries to connect them into as few as possible splines
428 void ShipCAD::JoinSplineSegments(float join_error,
429  bool force_to_one_segment,
430  SplineVector& list)
431 {
432  bool fixedclosed;
433  bool matchclosed;
434  Spline* nearest, *match;
435  // remove any single line segments
436  size_t i = 1;
437  while (i < list.size()) {
438  Spline* fixed = list.get(i-1);
439  if (fixed->numberOfPoints() > 1) {
440  fixedclosed = fixed->getFirstPoint().distanceToPoint(fixed->getLastPoint()) < 1E-5;
441  }
442  else
443  fixedclosed = false;
444  if (fixed->numberOfPoints() > 1 && !fixedclosed) {
445  float nearestdist = 1E30f;
446  nearest = 0;
447  for (size_t j=1; j<=list.size(); ++j) {
448  if (i == j)
449  continue;
450  match = list.get(j-1);
451  if (match->numberOfPoints() > 1)
452  matchclosed = match->getFirstPoint().distanceToPoint(match->getLastPoint()) < 1E-5;
453  else
454  matchclosed = false;
455  if (match->numberOfPoints() > 1 && !matchclosed) {
456  float d1 = SquaredDistPP(fixed->getFirstPoint(), match->getFirstPoint());
457  float d2 = SquaredDistPP(fixed->getFirstPoint(), match->getLastPoint());
458  float d3 = SquaredDistPP(fixed->getLastPoint(), match->getFirstPoint());
459  float d4 = SquaredDistPP(fixed->getLastPoint(), match->getLastPoint());
460  float min = Minimum(d1, d2, d3, d4);
461  if (min < nearestdist) {
462  nearestdist = min;
463  nearest = match;
464  if (min < 1E-5)
465  break;
466  }
467  }
468  }
469  if (nearest != 0) {
470  match = nearest;
471  float d1 = SquaredDistPP(fixed->getFirstPoint(), match->getFirstPoint());
472  float d2 = SquaredDistPP(fixed->getFirstPoint(), match->getLastPoint());
473  float d3 = SquaredDistPP(fixed->getLastPoint(), match->getFirstPoint());
474  float d4 = SquaredDistPP(fixed->getLastPoint(), match->getLastPoint());
475  float min = Minimum(d1, d2, d3, d4);
476  if (min < join_error || force_to_one_segment) {
477  // the splines do touch each other on one of their ends
478  if (min == d1)
479  fixed->insert_spline(0, true, true, *match);
480  else if (min == d2)
481  fixed->insert_spline(0, false, true, *match);
482  else if (min == d3)
483  fixed->insert_spline(fixed->numberOfPoints(), false, true, *match);
484  else if (min == d4)
485  fixed->insert_spline(fixed->numberOfPoints(), true, true, *match);
486  else
487  throw runtime_error("Error in comparing minimum values JoinSplineSegments");
488  list.del(match);
489  i = 0; // reset match index to start searching for a new matching spline
490  }
491  }
492  }
493  ++i;
494  }
495 }
