303 // keep track of drag state as fallows:
304 // _drag_state = 0 no dragging, move or pick in progress
305 // _drag_state = 1 button has been pressed, not moved far enough for a drag
306 // _drag_state = 2 button has been pressed, and dragging
307 // _drag_state = 3 left button has been pressed, dragging point. not moved far enough for a drag
308 // _drag_state = 4 left button has been pressed, dragging point
309 // _drag_state = 5 left button pressed, multi selecting, got a point on press
310 void
311 Viewport::mousePressEvent(QMouseEvent *event)
312 {
313 // cout << "mouse press enter:" << _drag_state << endl;
314  _drag_state = 1;
315  // on a left mouse press, check for selection
316  if (event->buttons().testFlag(Qt::LeftButton)) {
317  cout << "left mouse" << endl;
318  // check for shift-alt to create a flowline
319  if (event->modifiers() == (Qt::AltModifier | Qt::ShiftModifier) && getViewportType() != fvPerspective) {
320  cout << "shift-alt" << endl;
321  // convert mouse position back to 2D world coordinates
322  QVector2D pos = _view->projectTo3D(event->pos(), width(), height());
323  _ctl->addFlowline(pos, *this);
324  } else {
325  // find multi select, and only pick points
326  PickRay ray(event->modifiers() == Qt::ControlModifier, true, false, false);
327  if (canPick() && _view->leftMousePick(event->pos(), width(), height(), ray)) {
328  cout << "pick state changed - multisel:" << ray.multi_sel << endl;
329  update();
330  if (ray.multi_sel)
331  _drag_state = 5;
332  else
333  _drag_state = 3;
334  }
335  }
336  }
337  _prev_pos = event->pos();
338  _prev_buttons = event->buttons();
339  _drag_start = event->pos();
340  // are we getting mouse clicks in the gl window?
341 // cout << "mouse press exit: " << _drag_state << " pos:" << event->pos().x() << "," << event->pos().y() << endl;
342 }

344 void
345 Viewport::mouseReleaseEvent(QMouseEvent *event)
346 {
347 // cout << "mouse release enter:" << _drag_state << endl;
348  bool view_changed = false;
349  // for mouse release, the button released won't be in the set
350  if (_prev_buttons.testFlag(Qt::LeftButton) && !event->buttons().testFlag(Qt::LeftButton)) {
351  if (_drag_state == 1) {
352  // no movement, or not far enough to drag
353  // a point could have been picked in mouse down, if not multi selecting
354  PickRay ray(event->modifiers() == Qt::ControlModifier, false, true, canPickFace());
355  if ((canPick() || canPickFace())
356  && _view->leftMousePick(event->pos(), width(), height(), ray)) {
357  cout << "pick on release: multisel:" << ray.multi_sel << endl;
358  update();
359  }
360  }
361  // drag state = 2, 3, 4, 5 no picking
362  // 5 is no picking as even though multiselect, we got it on mouse down
363  else {
364  // dragging view or point
365  view_changed = _view->leftMouseRelease(event->pos(), width(), height());
366  if (_drag_state == 4) // dragging point
367  getController()->stopMovePoint();
368  }
369  }
370  else if (_prev_buttons.testFlag(Qt::RightButton) && !event->buttons().testFlag(Qt::RightButton)) {
371  if (_drag_state == 1)
372  view_changed = contextMenu(event);
373  else
374  view_changed = _view->rightMouseRelease(event->pos(), width(), height());
375  }
376 
377  if (view_changed)
378  update();
379  _prev_buttons = event->buttons();
380  // are we getting mouse clicks in the gl window?
381 // cout << "mouse release exit: " << _drag_state << " pos:" << event->pos().x() << "," << event->pos().y() << endl;
382  _drag_state = 0;
383 }

385 void
386 Viewport::mouseMoveEvent(QMouseEvent *event)
387 {
388 // cout << "mouse move enter:" << _drag_state << endl;
389 
390  bool view_changed = false;
391 
392  // check for actual dragging
393  if (_drag_state == 0 || _drag_state == 5
394  || (_drag_state == 1 && (event->pos() - _drag_start).manhattanLength() < QApplication::startDragDistance())
395  || (_drag_state == 3 && (event->pos() - _drag_start).manhattanLength() < QApplication::startDragDistance()))
396  return;
397 
398  if (_drag_state != 2 && _drag_state != 4)
399  _drag_state = _drag_state == 1 ? 2 : 4;
400 
401  if (event->buttons().testFlag(Qt::LeftButton)) {
402  // are we dragging a point?
403  if (_drag_state == 4) {
404  QVector3D coordchg;
405  if (_view->pointDrag(event->pos(), width(), height(), coordchg)) {
406  // the point has been dragged
407  getController()->movePoint(coordchg);
408  }
409  }
410  else {
411  view_changed = _view->leftMouseMove(event->pos(), _prev_pos, width(), height());
412  }
413  }
414  else if (event->buttons().testFlag(Qt::MidButton)) {
415  view_changed = _view->middleMouseMove(event->pos(), _prev_pos, width(), height());
416  }
417  else if (event->buttons().testFlag(Qt::RightButton)) {
418  view_changed = _view->rightMouseMove(event->pos(), _prev_pos, width(), height());
419  }
420 
421  if (view_changed)
422  update();
423 
424  _prev_pos = event->pos();
425  // are we getting mouse clicks in the gl window?
426 // cout << "mouse move exit:" << _drag_state << " pos:" << event->pos().x() << "," << event->pos().y() << endl;
427 }
