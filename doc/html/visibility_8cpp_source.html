67 void Visibility::loadBinary(FileBuffer &source)
68 {
69  clear();
70  quint32 n;
71  source.load(n);
72  _model_view = static_cast<model_view_t>(n);
73  source.load(_show_control_net);
74  source.load(_show_interior_edges);
75  source.load(_show_stations);
76  source.load(_show_buttocks);
77  source.load(_show_waterlines);
78  source.load(_show_normals);
79  source.load(_show_grid);
80  source.load(_show_diagonals);
81  source.load(_show_markers);
82  source.load(_show_curvature);
83  source.load(_curvature_scale);
84  if (_owner->getFileVersion() >= fv195) {
85  source.load(_show_control_curves);
86  if (_owner->getFileVersion() >= fv210) {
87  source.load(_cursor_increment);
88  if (fabs(_cursor_increment)<1e-5)
89  _cursor_increment = 0.1f;
90  if (_owner->getFileVersion() >= fv220) {
91  source.load(_show_hydrostatic_data);
92  source.load(_show_hydro_displacement);
93  source.load(_show_hydro_lateral_area);
94  source.load(_show_hydro_sectional_areas);
95  source.load(_show_hydro_metacentric_height);
96  source.load(_show_hydro_lcf);
97  if (_owner->getFileVersion() >= fv250) {
98  source.load(_show_flowlines);
99  }
100  }
101  }
102  }
103 }
